44      /**
45       * Constructor.
46       * @param newTag promptgroup string tag
47       * @param newIndex index value for this entry
48       */
49      public DBDGuiGroupData(String newTag, int newIndex) {
50          tag = newTag;
51          index = newIndex;
52      }
53  
54      /**
55       * Getter for string tag.
56       * @return gui_group string tag
57       */
58      public String getTag() {
59          return tag;
60      }
61  
62      /**
63       * Getter for index value.
64       * @return gui_group index
65       */
66      public int getIndex() {
67          return index;
68      }
69  }
