57  public void addDevice(DBDDeviceData dd) {
58  	// key def. : <recordtype>+"/"+<device choice string>
59  	
60  	if (!devices.containsKey((dd.record_type+"/"+dd.choice_string)))
61  		devices.put((dd.record_type+"/"+dd.choice_string), dd);
62  	else
63  		Console.getInstance().println("Device "+dd.record_type+"/"+dd.choice_string+" already exists in DBD - ignoring this definition.");
64  }
65  /**
66   * This method was created in VisualAge.
67   * @param md com.cosylab.vdct.MenuData
68   */
69  public void addMenu(DBDMenuData md) {
70  	if (!menus.containsKey(md.getName()))
71  		menus.put(md.getName(), md);
72  	else
73  		Console.getInstance().println("Menu "+md.getName()+" already exists in DBD - ignoring this definition.");
74  }

79  public void addRecord(DBDRecordData rd) {
80  	if (!records.containsKey(rd.name))
81  		records.put(rd.name, rd);
82  	else
83  		Console.getInstance().println("Record "+rd.getName()+" already exists in DBD - ignoring this definition.");
84  }
85  /**
86   * Find gui_group tag in global table and return index.
87   *
88   * @param tag promptgroup tag string
89   * @return gui_group data object
90   */
91  public DBDGuiGroupData findOrAddGuiGroup(String tag) {
92      DBDGuiGroupData gg = gui_groups.get(tag);
93  	if (gg == null) {
94          gg = new DBDGuiGroupData(tag,gui_groups.size()+1);
95  		gui_groups.put(tag, gg);
96      }
97      return gg;
98  }

102  * @param db com.cosylab.vdct.db.DBData

104 public boolean consistencyCheck(com.cosylab.vdct.db.DBData db) {
105 	String illegalString;
106 	
107 	ConsoleInterface console = Console.getInstance();
108 	
109 	// No output if everything ok.
110 	//console.println();
111 	//console.print("o) Checking consistency with DBD file...");
112 	if (db==null) {
113 		console.println();
114 		console.println("\t No DB data defined.");
115 		console.println();
116 		return false;
117 	}
118 	
119 	boolean isOK = true;
120 	com.cosylab.vdct.db.DBRecordData dbRecord;
121 	DBDRecordData dbdRecord;
122 	Enumeration e2;
123 	com.cosylab.vdct.db.DBFieldData dbField;
124 	DBDFieldData dbdField;
125 	DBDMenuData menu;
126 	String dev;
127 	
128 	// records check
129 	Enumeration e = db.getRecordsV().elements();
130 	while (e.hasMoreElements()) {
131 		dbRecord = (com.cosylab.vdct.db.DBRecordData)(e.nextElement());
132 		dbdRecord = this.getDBDRecordData(dbRecord.getRecord_type());
133 		if (dbdRecord!=null) {
134 			
135 			// fields check (fields, menus, devices)
136 			e2 = dbRecord.getFieldsV().elements();
137 			while (e2.hasMoreElements()) {
138 				dbField = (com.cosylab.vdct.db.DBFieldData)(e2.nextElement());
139 				dbdField = dbdRecord.getDBDFieldData(dbField.getName());
140 				if (dbdField!=null) {
141 
142 					// case when visual data is applied on non-existing (defined) field
143 					// aloww this
144 					if (dbField.getValue() == null)
145 						continue;
146 
147 					// device check
148 					if (dbdField.getField_type() == DBDConstants.DBF_DEVICE) {
149 						dev = dbRecord.getRecord_type()+"/"+dbField.getValue();
150 						if (dbField.getValue().indexOf("$")!=-1) {
151 								console.println();
152 								console.print("\tWarning: Record '"+dbRecord.getName()+"', field '"+dbField.getName()+"':");
153 								console.print(" Value '"+dbField.getValue()+"' is not valid device -> template definition?...");
154 								dbField.setTemplate_def(true);
155 						}
156 						else if (this.getDBDDeviceData(dev) == null) {
157 							console.println();
158 							console.print("\tWarning: Record type '"+dbRecord.getRecord_type()+"', field '"+dbdField.getName()+"':");
159 							console.print(" Device '"+dev+"' is not defined DBD file.");
160 
161 							/* RT 12438 - if dbd not present, DTYP information is gone 
162 						    illegalString="# field("+dbField.getName()+",\""+dbField.getValue()+"\")";
163 							if (dbField.getComment()!=null)	illegalString=dbField.getComment()+"\n"+illegalString;
164 							dbField.setComment(illegalString);
165 							
166 							dbField.setValue("");
167 							*/
168 						}
169 					}
170 					// menu check
171 					else if (dbdField.getField_type() == DBDConstants.DBF_MENU) {
172 						menu = this.getDBDMenuData(dbdField.getMenu_name());
173 						if (menu==null) {
174 							isOK = false;
175 							console.println();
176 							console.print("\tRecord type '"+dbRecord.getRecord_type()+"', field '"+dbdField.getName()+"':");
177 							console.print(" Menu '"+dbdField.getMenu_name()+"' is not defined DBD file (DBD file error)...");
178 						}
179 						else {
180 							if (!menu.containsValue(dbField.getValue())) {
181 								console.println();
182 							/*	console.print("\t Warning: Value '"+dbField.getValue()+"' is not valid for menu '"+dbdField.getMenu_name()+". Using defaults...");
183 
184 								illegalString="# field("+dbField.getName()+",\""+dbField.getValue()+"\")";
185 								if (dbField.getComment()!=null)	illegalString=dbField.getComment()+"\n"+illegalString;
186 								dbField.setComment(illegalString);
187 							
188 								dbField.setValue("");*/
189 								
190 								console.print("\tWarning: Record '"+dbRecord.getName()+"', field '"+dbField.getName()+"':");			
191 								console.print(" Value '"+dbField.getValue()+"' is not valid for menu '"+dbdField.getMenu_name()+" -> template definition?...");
192 								dbField.setTemplate_def(true);
193 								
194 							}
195 						}
196 					}
197 		
198 					
199 				}
200 				else {
201 					isOK=false;
202 					console.println();
203 					console.print("\tRecord '"+dbRecord.getName()+"':");
204 					console.print(" Field '"+dbField.getName()+"' in record type '"+dbRecord.getRecord_type()+"' is not defined in DBD file. Field will be commented out when saved!");
205 					
206 					illegalString="# illegal line - undefined field: field("+dbField.getName()+",\""+dbField.getValue()+"\")";
207 					if (dbRecord.getComment()!=null) illegalString=dbRecord.getComment()+"\n"+illegalString;
208 					dbRecord.setComment(illegalString);
209 				}
210 			}
211 		}
212 		else {
213 			isOK=false;
214 			console.println();
215 			console.print("\tRecord '"+dbRecord.getName()+"':");
216 			console.print(" Record type '"+dbRecord.getRecord_type()+"' is not defined in DBD file.");
217 		}
218 	}
219 
220 	if (isOK) {
221 		// No output if everything ok.
222 		//console.println();
223 		//console.print("\tOK");
224 	} else {
225 		console.println();
226 		console.print("o) DB file is not consistent with DBD file!");
227 		console.println();
228 		console.println();
229 	}
230 	return isOK;
231 }
