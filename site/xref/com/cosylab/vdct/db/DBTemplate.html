31  import java.util.Hashtable;
32  import java.util.Vector;
33  
34  /**
35   * @author Matej
36   */
37  public class DBTemplate extends DBComment
38  {
39  	protected String id = null;
40  	protected String fileName = null;
41  	protected String description = null;
42  
43  	protected long modificationTime = 0; 
44  	
45  	protected String version = null;
46  	protected String ioc = null;
47  
48  	protected Hashtable ports = null;
49  	protected Vector portsV = null;
50  
51  	protected Hashtable macros = null;
52  	protected Vector macrosV = null;
53  
54  	protected boolean initialized;
55  
56  	// data
57  	protected DBData data = null;
58  
59  	/**
60  	 * Constructor.
61       * @param id
62       * @param fileName
63  	 */
64  	public DBTemplate(String id, String fileName)
65  	{
66  		this.id=id;
67  		this.fileName=fileName;
68  		modificationTime = 0;
69  
70  		ports = new Hashtable();
71  		portsV = new Vector();
72  
73  		macros = new Hashtable();
74  		macrosV = new Vector();
75  	}
76  	
77  
78  	/**
79  	 * Returns the description.
80  	 * @return String
81  	 */
82  	public String getDescription()
83  	{
84  		return description;
85  	}
86  
87  	/**
88  	 * Returns the fileName.
89  	 * @return String
90  	 */
91  	public String getFileName()
92  	{
93  		return fileName;
94  	}
95  
96  	/**
97  	 * Returns the id.
98  	 * @return String
99  	 */
100 	public String getId()
101 	{
102 		return id;
103 	}
104 
105 	/**
106 	 * Sets the description.
107 	 * @param description The description to set
108 	 */
109 	public void setDescription(String description)
110 	{
111 		this.description = description;
112 	}
113 
114 	/**
115 	 * Sets the fileName.
116 	 * @param fileName The fileName to set
117 	 */
118 	public void setFileName(String fileName)
119 	{
120 		this.fileName = fileName;
121 	}
122 
123 	/**
124 	 * Sets the id.
125 	 * @param id The id to set
126 	 */
127 	public void setId(String id)
128 	{
129 		this.id = id;
130 	}
131 
132 
133 	/**
134 	 * Sets the data.
135 	 * @param data The data to set
136 	 */
137 	public void setData(DBData data)
138 	{
139 		this.data = data;
140 	}
141 	
142 	/**
143 	 * @return the modificationTime
144 	 */
145 	public long getModificationTime() {
146 		return modificationTime;
147 	}
148 
149 	/**
150 	 * @param modificationTime the modificationTime to set
151 	 */
152 	public void setModificationTime(long modificationTime) {
153 		this.modificationTime = modificationTime;
154 	}
155 
156 	/**
157 	 * Returns the initialized.
158 	 * @return boolean
159 	 */
160 	public boolean isInitialized()
161 	{
162 		return initialized;
163 	}
164 
165 	/**
166 	 * Sets the initialized.
167 	 * @param initialized The initialized to set
168 	 */
169 	public void setInitialized(boolean initialized)
170 	{
171 		this.initialized = initialized;
172 	}
173 
174 	/**
175 	 * Returns the ports.
176 	 * @return Hashtable
177 	 */
178 	public Hashtable getPorts()
179 	{
180 		return ports;
181 	}
182 
183 	/**
184 	 * Returns the portsV.
185 	 * @return Vector
186 	 */
187 	public Vector getPortsV()
188 	{
189 		return portsV;
190 	}
191 
192 	/**
193      * @param port
194 	 */
195 	public void addPort(DBPort port)
196 	{
197 		if (!ports.containsKey(port.getName()))
198 		{
199 			ports.put(port.getName(), port);
200 			portsV.addElement(port);
201 		}
202 	}
203 
204 	/**
205      * @param macro
206 	 */
207 	public void addMacro(DBMacro macro)
208 	{
209 		if (!macros.containsKey(macro.getName()))
210 		{
211 			macros.put(macro.getName(), macro);
212 			macrosV.addElement(macro);
213 		}
214 	}
215 
216 	/**
217 	 * Returns the data.
218 	 * @return DBData
219 	 */
220 	public DBData getData()
221 	{
222 		return data;
223 	}
224 
225 	/**
226 	 * Returns the macros.
227 	 * @return Hashtable
228 	 */
229 	public Hashtable getMacros()
230 	{
231 		return macros;
232 	}
233 
234 	/**
235 	 * Returns the macrosV.
236 	 * @return Vector
237 	 */
238 	public Vector getMacrosV()
239 	{
240 		return macrosV;
241 	}
242 
243 	public String getVersion() {
244 		return version;
245 	}
246 
247 	public void setVersion(String version) {
248 		this.version = version;
249 	}
250 
251 	public String getIoc() {
252 		return ioc;
253 	}
254 
255 	public void setIoc(String ioc) {
256 		this.ioc = ioc;
257 	}
258 }
