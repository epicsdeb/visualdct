185 /**
186  * Check is DTYP field is defined before INP and OUT fields...
187  * Insert the method's description here.
188  * Creation date: (18.11.1999 18:26:27)
189  * @param db
190  * @param dbd

192 
193 public static void checkDTYPfield(DBData db, DBDData dbd) {
194 
195  DBDRecordData dbdRecord;
196 
197  DBRecordData dbRecord;
198  Enumeration e = db.getRecordsV().elements();
199  Enumeration e2;
200  while (e.hasMoreElements()) {
201 	dbRecord = (DBRecordData)e.nextElement();
202 	dbdRecord = dbd.getDBDRecordData(dbRecord.getRecord_type());
203 	if (dbdRecord!=null) {
204 		
205 		DBFieldData dbField2 = null;
206 		DBFieldData dbField = (DBFieldData)dbRecord.getFields().get("DTYP");
207 		if (dbField!=null) {
208 			e2 = dbRecord.getFieldsV().elements();
209 			dbField2 = (DBFieldData)e2.nextElement();
210 			while (dbField!=dbField2) {
211 				if (dbField2.getName().equals("INP") || dbField2.getName().equals("OUT")) 
212 					break;
213 				else
214 					dbField2 = (DBFieldData)e2.nextElement();
215 			}
216 
217 			if (dbField!=dbField2) 
218 				Console.getInstance().println("Warning: "+dbRecord.name+" -> DTYP field needs to be defined before two fields INP and OUT...");
219 		}
220 
221 	}
222  }
