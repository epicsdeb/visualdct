24  public class Help {
25      
26      public static final String RUN_COMMAND_CLASS = "java -cp VisualDCT.jar com.cosylab.vdct.VisualDCT";
27      public static final String RUN_COMMAND_JAR = "java -jar VisualDCT.jar";
28      public static final String DBD_LOAD = "<DBD>* or <DB>*";
29      public static final String DIR = "-DVDCT_DIR=";
30      public static final String EPICS_DB_INCLUDE_PATH = "-DEPICS_DB_INCLUDE_PATH=";
31      public static final String CONFIG_DIR = "-DVDCT_CONFIG_DIR=";
32      public static final String HELP = "-h";
33      public static final String HELP2 = "--help";
34      public static final String PATH = "<path>";
35      public static final String SPACE = "     ";
36      public static final String INDENT = "  ";
37        
38      public static final String GENERATE_FLAT_DATABASE = "java -cp VisualDCT.jar com.cosylab.vdct.GenerateFlatDatabase";
39      public static final String OPTIONS = "[OPTIONS]";
40      public static final String INPUT_VDB = "input.vdb";
41      public static final String OUTPUT_DB = "output.db";
42      public static final String FLATDB = "flatdb";
43      public static final String DBD_FILE = "--dbd-file";
44      public static final String ENABLE_GLOBAL_MACRO = "--enable-global-macros";
45      public static final String DISABLE_GLOBAL_MACRO = "--disable-global-macros";
46      public static final String ENABLE_CAPFAST = "--enable-capfast";
47      public static final String DISABLE_CAPFAST ="--disable-capfast";
48      
49      public static void printHelpAndExit() {
50          System.out.println("VisualDCT help");
51          System.out.println();
52          System.out.println("General help");
53          System.out.println("Usage: " + RUN_COMMAND_CLASS + " " + OPTIONS);
54          System.out.println("or     " + RUN_COMMAND_JAR + " " + OPTIONS);
55          System.out.println();
56          System.out.println("Options:");
57          System.out.println(INDENT + HELP + ", " + HELP2 + "                    " + SPACE +
58                  "print this message");
59          System.out.println(INDENT + DBD_LOAD + "               " + SPACE +
60                  "paths to database definition files and record instance database files (both optional)");
61          System.out.println(INDENT + DIR + PATH + "             " + SPACE +
62                  "VDCT_DIR environment variable is used to define the default working directory");
63          System.out.println(INDENT + CONFIG_DIR + PATH + "      " + SPACE + 
64                  "VDCT_CONFIG_DIR is an environment variable used to define the default plug-ins configuration directory");
65          System.out.println(INDENT + EPICS_DB_INCLUDE_PATH + PATH + SPACE + 
66                  "definition of EPICS_DB_INCLUDE_PATH environment variable");
67          System.out.println();
68          System.out.println(INDENT + "Example: " + RUN_COMMAND_CLASS + " " + CONFIG_DIR + "/users/devl/config/vdct test.dbd test.db" );
69          
70          System.out.println();
71          System.out.println("Generating flat database help");
72          System.out.println("Usage: " + GENERATE_FLAT_DATABASE + " " + OPTIONS + " ..." + " " + INPUT_VDB + " " + OUTPUT_DB);
73          System.out.println();
74          System.out.println("Options:");
75          System.out.println(INDENT + DBD_FILE + "                    " + SPACE + 
76                  "if this command is followed by the name of a .dbd file, the .dbd file is loaded before database is generated");
77          System.out.println(INDENT + ENABLE_GLOBAL_MACRO + "        " + SPACE + 
78                  "enables global macro evaluation");
79          System.out.println(INDENT + DISABLE_GLOBAL_MACRO + "       " + SPACE +
80                  "disable global macro evaluation");
81          System.out.println(INDENT + ENABLE_CAPFAST + "              " + SPACE +
82                  "enable production of hierarhical names like CapFast");
83          System.out.println(INDENT + DISABLE_CAPFAST + "             " + SPACE +
84                  "disable production of hierarhical names like CapFast");
85          System.out.println();
86          System.out.println("We also provided a script for running flattening database tool, named flatdb, which can be found next to the distribution of VisualDCT.");
87          System.out.println(INDENT + "Example: " + FLATDB + " " + DBD_FILE + " app.dbd " + INPUT_VDB + " " + OUTPUT_DB);
88          
89          
90          
91          System.exit(0);
92      }
