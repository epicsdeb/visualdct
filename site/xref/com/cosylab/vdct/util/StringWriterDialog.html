40  	private void init() {
41  		jTextField.getDocument().addDocumentListener(new DocumentListener() {
42  			private DBDEntry entry = new DBDEntry("");
43  
44  			public void changedUpdate(DocumentEvent e) {
45  				// we won't ever get this with a PlainDocument
46  				
47  			}
48  
49  			public void insertUpdate(DocumentEvent e) {
50  				update(e);
51  			}
52  
53  			public void removeUpdate(DocumentEvent e) {
54  				update(e);
55  			}
56  			
57  			private void update(DocumentEvent e) {
58  				entry.setValue(jTextField.getText());				
59  				jLabelValue.setText(DBDEntry.matchAndReplace(jTextField.getText()));
60  				boolean ok = entry.getFile().canRead() && entry.getFile().isFile();
61  				jLabelWarning.setVisible(!ok);
62  				jButtonDefault.setEnabled(ok);
63  			}
64  		});
65  
66  		jComboBoxMacros.setModel(new ComboBoxModel() {
67  			Vector elements = null;
68  			Object selectedItem;
69  		
70  			public Object getSelectedItem() {
71  				if (elements==null) loadElements();
72  				return selectedItem;
73  			}
74  
75  			public void setSelectedItem(Object anItem) {
76  				selectedItem = anItem;
77  			}
78  
79  			public int getSize() {
80  				if (elements==null) loadElements();
81  				return elements.size();
82  			}
83  
84  			public Object getElementAt(int index) {
85  			   	if (elements==null) loadElements();
86  				return elements.get(index);
87  			}
88  
89  			private void loadElements() {
90  				Enumeration e = DBDEntry.getProperties().propertyNames();
91  				elements = new Vector();
92  				while (e.hasMoreElements()) {
93  					elements.add(e.nextElement());
94  				}
95  				Collections.sort(elements);
96  				selectedItem = elements.get(0);
97  			}
98  
99  			public void addListDataListener(ListDataListener l) {
100 				
101 			}
102 
103 			public void removeListDataListener(ListDataListener l) {
104 				
105 			}
106 			
107 		});	
108 		
109 		jButtonDefault.setEnabled(false);	
110 	}
