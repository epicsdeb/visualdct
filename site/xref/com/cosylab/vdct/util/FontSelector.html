32  import java.awt.*;
33  import java.awt.event.*;

35  public class FontSelector extends JPanel
36  {
37  	private static final String nullString = "";
38  	private static final String exampleText = "The quick brown fox jumps over a lazy dog.";
39  	private static final String plainStyleString = "Plain";
40  	private static final String boldStyleString = "Bold";
41  	private static final String italicStyleString = "Italic";
42  	private static final String boldAndItalicStyleString = "Bold&Italic";
43  	
44  
45      FontDisplayPanel displayPanel;
46      JComboBox fontsBox, fontStylesBox, fontSizesBox;
47  
48      String[] fontStyleLabels = {plainStyleString, boldStyleString, italicStyleString, boldAndItalicStyleString};
49      int BOLDITALIC = Font.BOLD|Font.ITALIC;
50      int[] fontStyles = {Font.PLAIN, Font.BOLD, Font.ITALIC, BOLDITALIC};
51      String[] fontSizeLabels = { "8", "9", 
52                                 "10", "11",
53                                 "12", "14",
54                                 "18", "20",
55                                 "22", "24",
56                                 "26", "28",
57                                 "36", "48", 
58                                 "72"};
59                                 
60  	public FontSelector()
61  	{
62  		super();
63  		
64  		init();
65  	}

67      public void init() {
68  
69          // 2. Create a display panel and add it to the applet ....
70  
71          displayPanel = new FontDisplayPanel();
72  
73  		setLayout(new GridLayout(2, 1));
74  
75          // 3. Create a control panel with titled border
76  
77          JPanel controlPanel = new JPanel();
78  
79          add(BorderLayout.NORTH, controlPanel);
80  
81          // 4. Create combo boxes for font names, styles, and sizes
82  
83          fontsBox = new JComboBox(displayPanel.fontFamilyNames);
84          fontsBox.setSelectedItem("Arial"); // default selection
85          fontsBox.addActionListener(new ComboBoxListener());
86  
87          fontStylesBox = new JComboBox(fontStyleLabels);
88          fontStylesBox.addActionListener(new ComboBoxListener());
89  
90          fontSizesBox = new JComboBox(fontSizeLabels);
91          fontSizesBox.setSelectedItem("12");
92          fontSizesBox.addActionListener(new ComboBoxListener());
93  
94          // 5. Add the combo boxes to the control panel and add the
95          //    panel to the main frame.
96  
97          controlPanel.add(fontsBox);
98          controlPanel.add(fontStylesBox);
99          controlPanel.add(fontSizesBox);
100 
101         displayPanel = new FontDisplayPanel();
102 		JScrollPane scrollPane = new JScrollPane(displayPanel);
103 
104         add(BorderLayout.NORTH, controlPanel);
105         add(scrollPane);
106     }
107 
108     // 6. Combo box listener to handle font name, style and size
109     //    selections in the respective combo boxes.
110 
111     class ComboBoxListener implements ActionListener {
112         public void actionPerformed(ActionEvent e) {
113             JComboBox tempBox = (JComboBox) e.getSource();
114 
115             if (tempBox.equals(fontsBox)) {
116                 displayPanel.fontFamilyName =
117                              (String) tempBox.getSelectedItem();
118                 displayPanel.repaint();
119             }
120             else if (tempBox.equals(fontStylesBox)) {
121                 displayPanel.fontStyle =
122                              fontStyles[tempBox.getSelectedIndex()];
123                 displayPanel.repaint();
124             }
125             else if (tempBox.equals(fontSizesBox)) {
126                 displayPanel.fontSize =
127                              Integer.parseInt((String) tempBox.getSelectedItem());
128                 displayPanel.repaint();
129             }
130         }
131     }

133     // 7. Definition of display panel
134 
135     public class FontDisplayPanel extends JPanel {
136         String displayText;
137         Font currentFont;
138         String fontFamilyName;
139         int fontStyle;
140         int fontSize;
141 
142         GraphicsEnvironment ge;
143         String[] fontFamilyNames;
144         
145         // 8. Constructor for display panel....
146 
147         public FontDisplayPanel() {
148 
149             // Logo to be displayed on the panel
150 
151             displayText = exampleText;
152             fontFamilyName = "Arial";
153             fontStyle = Font.PLAIN;
154             fontSize = 12;
155             
156             // Retrieve all the font family names from the platform
157 
158 //            System.out.println("Loading Fonts... Please Wait...");
159             ge = GraphicsEnvironment.getLocalGraphicsEnvironment();
160             fontFamilyNames = ge.getAvailableFontFamilyNames();
161             setBackground(Color.white);  // For canvas background color
162         }
163 
164         // 9. The update method...
165 
166         public void update(Graphics g) {
167             g.clearRect(0, 0, getWidth(), getHeight());
168             paintComponent(g);
169         }
170 
171         // 10. The painting method...
172 
173         public void paintComponent(Graphics g) {
174             super.paintComponent(g);
175 
176             // 11. Create the graphics context object
177 
178             Graphics2D g2D = (Graphics2D) g;
179 			//Graphics g2D = g;
180 
181             // 12. Prepare the current font and apply it to
182             //     the display text.
183 
184             currentFont = new Font(fontFamilyName,
185                                    fontStyle,
186                                    fontSize);
187             g2D.setFont(currentFont);
188             
189             g2D.setColor(getForeground());
190            
191 			Dimension dimension = getSize();
192 			
193 			if(displayText.compareTo(nullString) == 0)
194 				displayText = exampleText;
195 
196             g2D.drawString(displayText, 16, dimension.height - 16);
197         }
198         
199 		public void setDisplayText(String parDisplayText)
200 		{
201 			displayText = parDisplayText;
202 		}
203     }
204 
205 	public void setFont(Font parFont)
206 	{
207 		super.setFont(parFont);
208 
209 		if(fontsBox != null)
210 			fontsBox.setSelectedItem(parFont.getFamily());
211 
212 		if(fontStylesBox != null)
213 		{
214 			int style = parFont.getStyle();
215 			
216 			if(style == Font.PLAIN)
217 		    	fontStylesBox.setSelectedItem(plainStyleString);
218 
219 			else if(style == Font.BOLD)
220 		    	fontStylesBox.setSelectedItem(boldStyleString);
221 
222 			else if(style == Font.ITALIC)
223 		    	fontStylesBox.setSelectedItem(italicStyleString);
224 
225 			else if(style == BOLDITALIC)
226 		    	fontStylesBox.setSelectedItem(boldAndItalicStyleString);
227 		}
228 
229 		if(fontSizesBox != null)
230 		    fontSizesBox.setSelectedItem(String.valueOf(parFont.getSize()));
231 			
232 		if(displayPanel != null)
233 		{
234 			displayPanel.currentFont = parFont;
235        	
236 			displayPanel.fontFamilyName = parFont.getFamily();
237 			displayPanel.fontStyle = parFont.getStyle();
238 			displayPanel.fontSize = parFont.getSize();
239 		}
240 	}
241         
242 	public Font getFont()
243 	{
244 		if(displayPanel != null)
245 			return displayPanel.currentFont;
246 
247 		return super.getFont();
248 	}
249 
250 	public FontDisplayPanel getDisplayPanel()
251 	{
252 		return displayPanel;
253 	}
