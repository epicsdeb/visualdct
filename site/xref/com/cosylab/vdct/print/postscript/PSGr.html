151 /** Creates a new PSGr Object that is a copy of the original PSGr Object.  */

153 /** Container.paint calls create and dispose,
154  * for a separate g to clip and position each child
155  * and then restore larger clip and original coordinates
156  * by disposing of the smaller g and using the original g.
157  * We have our own Container painting code,
158  * which works the same way.
159  * This simulates the context stack in PostScript,
160  * and works fine as long as each g is used as described,
161  * that is by using only the most recently created g.
162      * @return something
163  */
164 
165     public Graphics create () {
166         emitThis ("% create");
167         PSGr psgr = new PSGr (os,/*realg,*/CLONE);
168         psgr.font = font;
169         psgr.clippingRect = clippingRect;
170         psgr.clr = clr;
171         psgr.backClr = backClr;
172         emitThis ("% create => "+psgr);
173         return (Graphics) psgr;
174     }

176 /**
177  * Creates a new PSGr Object with the specified parameters,
178  * based on the original PSGr Object.
179  * This method translates the origin to the specified x and y,
180  * and then clips to the area with specified width and height.
181  * The resulting clipping area is the intersection
182  * of the current clipping area and the specified rectangle.
183  *
184  * We first "gsave" the PostScript context here,
185  * and we "grestore" it when dispose () is called,
186  * so if create () and dispose () are used in pairs,
187  * the earlier and larger clip should be preserved,
188  * and what the old g thinks is the clip should be so.
189  * NOT TESTED YET!
190  *
191  *** Compiler won't let this one be PSGr only Graphics
192  *** I think because it is part of the interface spec.
193  *
194  * @param x the x coordinate, pixels positive right from left.
195  * @param y the y coordinate, pixels positive down from top.
196  * @param width the width of the area
197  * @param height the height of the area
198  * @return something
199  * @see #translate
200  */
201     public Graphics create (int x, int y, int width, int height) {
202         emitThis ("% create xywh "+x+" "+y+" "+width+" "+height);
203         Graphics cg = create ();
204 
205         emitThis ("gsave");
206 
207         cg.translate (x, y);
208         cg.clipRect (0, 0, width, height);
209 
210         return cg;
211     }

372  * @see #drawChars

374     public void setFont (Font f) {
375         emitThis ("%setFont");
376         if (f != null) {
377             this.font = f;
378             String javaName = font.getName ();
379             int javaStyle = font.getStyle ();
380             String psName;
381 
382             if (javaName.equals("Symbol"))
383                 psName = "Symbol";
384 
385 /* Times says Italic... */
386 
387             else if (javaName.startsWith("Times")
388               || javaName.equals ("Serif")) {
389                 psName = "Times-";
390                 switch (javaStyle) {
391                 case Font.PLAIN:
392                     psName += "Roman";
393                     break;
394                 case Font.BOLD:
395                     psName += "Bold";
396                     break;
397                 case Font.ITALIC:
398                     psName += "Italic";
399                     break;
400                 case (Font.ITALIC + Font.BOLD):
401                     psName += "BoldItalic";
402                     break;
403                 }
404             }
405 
406 /* ...but Helvetica and Courier say Oblique */
407 
408             else if (javaName.equals("Helvetica")
409               || javaName.equals ("Courier")) {
410                 psName = javaName;
411                 switch (javaStyle) {
412                 case Font.PLAIN:
413                     break;
414                 case Font.BOLD:
415                     psName += "-Bold";
416                     break;
417                 case Font.ITALIC:
418                     psName += "-Oblique";
419                     break;
420                 case (Font.ITALIC + Font.BOLD):
421                     psName += "BoldOblique";
422                     break;
423                 }
424             }
425 
426 /* Just more names for Helvetica */
427 
428             else if (javaName.equals("SansSerif")
429               || javaName.equals ("Dialog")) {
430                 psName = "Helvetica";
431                 switch (javaStyle) {
432                 case Font.PLAIN:
433                     break;
434                 case Font.BOLD:
435                     psName += "-Bold";
436                     break;
437                 case Font.ITALIC:
438                     psName += "-Oblique";
439                     break;
440                 case (Font.ITALIC + Font.BOLD):
441                     psName += "BoldOblique";
442                     break;
443                 }
444             }
445 
446 /* And for Courier */
447 
448             else if (javaName.equals("Monospaced")) {
449                 psName = "Courier";
450                 switch (javaStyle) {
451                 case Font.PLAIN:
452                     break;
453                 case Font.BOLD:
454                     psName += "-Bold";
455                     break;
456                 case Font.ITALIC:
457                     psName += "-Oblique";
458                     break;
459                 case (Font.ITALIC + Font.BOLD):
460                     psName += "BoldOblique";
461                     break;
462                 }
463             }
464 
465             else
466                 psName = "Courier";
467 
468             emitThis ("/" + psName + " findfont");
469             emitThis (""+font.getSize()+" scalefont setfont");
470         }

551     public void setClip (int x, int y, int width, int height) {
552         emitThis ("%setClip "+x+" "+y+" "+width+" "+height);

561         //// emitThis ("initclip");
562         emitThis ("grestoreall");

564         emitThis ("newpath");

566         emitThis (xps + " " + yps + " moveto");

1059      * @param x x
1060      * @param y y

1069     protected boolean doImage (Image img,
1070     int x, int y, int width, int height,
1071       ImageObserver observer,
1072       Color bgcolor) {
1073 
1074         String bgString = null;
1075 
1076         if (null != bgcolor)
1077 	    bgString =
1078 	     myHexString (bgcolor.getRed ())
1079 	    +myHexString (bgcolor.getGreen ())
1080 	    +myHexString (bgcolor.getBlue ());
1081 
1082         int yps = yTrans (y);
1083         int xps = xTrans (x);
1084         int wps = dTrans (width);
1085         int hps = dTrans (height);
1086 
1087 /* Get image dimensions and data */
1088 
1089 	int iw = img.getWidth (observer);
1090 	int ih = img.getHeight (observer);
1091 
1092 /* Punt --
1093  * should just wait here until we have it,
1094  * waiting for image before coming here is the safest way */
1095 
1096 	if ((0 > iw) || (0 > ih))
1097 	    return false;
1098 
1099         emitThis ("gsave");
1100 
1101      //// emitThis ("20 dict begin");
1102 
1103 /* We need not write and read exact scanlines,
1104  * or even write and read the same size lines,
1105  * as long as we come out even by the end.
1106  * Doing so keeps the code simple,
1107  * but writing very long lines may be a problem elsewhere,
1108  * e.g. emailing a file may truncate long lines. */
1109 
1110         emitThis ("% string to hold a scanline's worth of data");
1111         emitThis ("/pix "+(iw*6)+" string def");
1112 
1113         emitThis ("% space for color conversion to gray");
1114         emitThis ("/grays "+(iw*2)+" string def");
1115         emitThis ("/npixls 0 def");
1116         emitThis ("/rgbindx 0 def");
1117 
1118 /* if calling width or height is 0,
1119  * it means to use actual image dimensions,
1120  * otherwise scale to fill given rectangle. */
1121 
1122         if (hps == 0 || wps == 0) {
1123             hps = ih;
1124             wps = iw;
1125         }
1126 
1127 /* lower left corner */
1128 
1129         emitThis (xps+" "+(yps-hps)+" translate");
1130 
1131 /* scale from unit square to size to be filled */
1132 
1133         emitThis (wps+" "+hps+" scale");
1134 
1135 /* width height bits */
1136 
1137         emitThis (iw+" "+ih+" 8");
1138 
1139 /* Matrix [w 0 0 -h 0 h]
1140  * for image scanned left-to-right and top-down as Java does
1141  * to fill a unit square */
1142 
1143         emitThis ("["+iw+" 0 0 -"+ih+" 0 "+ih+"]");
1144 
1145 /* Procedure to read following data lines --
1146  * discarding the EOF test because it would never happen,
1147  * instead would read beyond data into following code! */
1148 
1149         emitThis ("{currentfile pix readhexstring pop}");
1150 
1151 /* last couple of args for the Level 2 operator,
1152  * single stream and 3 color channels */
1153 
1154         emitThis ("false 3");
1155 
1156 /* Do it (or fake it) */
1157 
1158         emitThis ("colorimage");
1159 
1160 /* Blank cosmetic line before the data lines --
1161  * harmless with readhexstring */
1162 
1163         emitThis ("");
1164 
1165 	int nchars = 0;
1166 
1167 /* This is simplest but BAD FOR LARGE IMAGES,
1168  * because it requires a lot of memory,
1169  * better to grab and emit by bands.
1170  * Although we could use the same pixels array for each band,
1171  * it seems we would need a new PixelGrabber each time,
1172  * because the constructor requires x y w h. */
1173 
1174 	int [] pixels = new int [iw * ih];
1175 
1176         PixelGrabber pg = new PixelGrabber (img, 0, 0, iw, ih, pixels, 0, iw);
1177 
1178 /* Here again simplest but BAD FOR LARGE IMAGES,
1179  * better use grabPixels (maxMSec)
1180  * note that either form throws InterruptedException.
1181  * Should never use wait-forever. */
1182 
1183 	try { pg.grabPixels (); }
1184 	catch (Exception ex) {
1185 	    System.out.println ("Failed grabPixels");
1186 	    return false;
1187 	}
1188 
1189 	for (int py = 0; py < ih; py++) {
1190 	    for (int px = 0; px < iw; px++) {
1191 		int apixel = pixels [py * iw + px];
1192 
1193 		int alpha = (apixel >> 24) & 0xff;
1194 		int red   = (apixel >> 16) & 0xff;
1195 		int green = (apixel >>  8) & 0xff;
1196 		int blue  = (apixel ) & 0xff;
1197 
1198 /* Completely transparent pixels get BG if one specified.
1199  * We make no other (partial) use of alpha,
1200  * either we paint BG or stated color,
1201  * never leave the PostScript BG. */
1202 
1203 		if ((0 == alpha) && (null != bgString))
1204 		    emitThisNext (bgString);
1205 
1206 		else
1207 		    emitThisNext
1208 		    (myHexString (red)
1209 		    +myHexString (green)
1210 		    +myHexString (blue));
1211 
1212 /* Because readhextstring ignores whitespace,
1213  * we can write these safe and tidy blocks regardless,
1214  * and still use a read buffer sized to image width. */
1215 
1216 		nchars +=6;
1217 		if (60 == nchars) {
1218 		    emitThis ("");
1219 		    nchars = 0;
1220 		}
1221 	    }
1222 	}
1223 	if (0 < nchars)
1224 	    emitThis ("");
1225 
1226 /* Blank cosmetic line after all the data */
1227 
1228         emitThis ("");
1229 
1230 /* Done with our dict, clean up state */
1231 
1232      //// emitThis ("end");
1233         emitThis ("grestore");
1234 
1235         return true;
1236     }

1694 /** Find scale factor that will fit the page,
1695  * use smaller of x and y for both,
1696  * to convert all pixels to points.
1697  * Only reduce scale if needed -- do not ever magnify.
1698  * PostScript scale operator also changes text sizes.
1699  * Must allow for non-printing margins all around.
1700  * @param top top
1701  */
1702 
1703     public void scalePaint (Component top) {
1704 
1705         Dimension topSize = top.getSize ();
1706 
1707         float xScale = ((float)Page_Width-2*Page_MarginX)
1708             / ((float)topSize.width);
1709         float yScale = ((float)Page_Height-2*Page_MarginY)
1710             / ((float)topSize.height);
1711 
1712         float xyScale = yScale;
1713 
1714         if (xScale < yScale)
1715             xyScale = xScale;
1716 
1717         if (xyScale > 1)
1718             xyScale = 1;
1719 
1720         emitThis (xyScale+" "+xyScale+" scale");
1721     }
