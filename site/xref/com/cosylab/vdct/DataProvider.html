121 private void loadDefaultLinkTypeConfig()
122 {
123 	// pattern, default value, description
124 	linkTypeConfigTable.put("CONSTANT", new Object[] { Pattern.compile(".*"), "", "CONSTANT" });
125 	linkTypeConfigTable.put("PV_LINK", new Object[] { Pattern.compile(".*"), "", "PV_LINK" } );
126 	linkTypeConfigTable.put("VME_IO", new Object[] { Pattern.compile("#C\\d+ S\\d+( )*( @.*)?"), "#C0 S0 @", "VME_IO - #Ccard Ssignal @parm" });
127 	linkTypeConfigTable.put("CAMAC_IO", new Object[] { Pattern.compile("#B\\d+ C\\d+ N\\d+ A\\d+ F\\d+( )*( @.*)?"), "CAMAC_IO - #B0 C0 N0 A0 F0 @", "#Bbranch Ccrate Nstation Asubaddress Ffunction @parm" });
128 	linkTypeConfigTable.put("AB_IO", new Object[] { Pattern.compile("#L\\d+ A\\d+ C\\d+ S\\d+( )*( @.*)?"), "#L0 A0 C0 S0 @", "AB_IO - #Llink Aadapter Ccard Ssignal @parm" });
129 	linkTypeConfigTable.put("GPIB_IO", new Object[] { Pattern.compile("#L\\d+ A\\d+( )*( @.*)?"), "#L0 A0 @", "GPIB_IO - #Llink Aaddr @parm" });
130 	linkTypeConfigTable.put("BITBUS_IO", new Object[] { Pattern.compile("#L\\d+ N\\d+ P\\d+ S\\d+( )*( @.*)?"), "BITBUS_IO - @L0 N0 P0 S0 @", "#Llink Nnode Pport Ssignal @parm" });
131 	linkTypeConfigTable.put("INST_IO", new Object[] { Pattern.compile("@.*"), "@", "INST_IO - @" });
132 	linkTypeConfigTable.put("BBGPIB_IO", new Object[] { Pattern.compile("#L\\d+ B\\d+ G\\d+( )*( @.*)?"), "#L0 B0 G0 @", "BBGPIB_IO - #Llink Bbbaddr Ggpibaddr @parm" });

150 public void addInspectableListener(InspectableObjectsListener listener) {
151 	Vector listeners = (Vector)inspectableListeners.get(listener.getDsId());
152 	if (listeners == null) {
153 		listeners = new Vector();
154 		inspectableListeners.put(listener.getDsId(), listeners);
155 	}
156 	
157 	if (!listeners.contains(listener)) {
158 		listeners.addElement(listener);
159 	}
160 }

210      *
211      * @return something
212      */
213     public Vector getInspectable() {
214 	Vector objs = new Vector();
215 	Iterator iterator = Group.getAllRoots().iterator();
216 	while (iterator.hasNext()) {
217 		objs.addAll(getInspectable(((Group)iterator.next()).getDsId()));
218 	}
219 	return objs;
220 }
221 /**
222  * Insert the method's description here.
223  * Creation date: (2.2.2001 21:59:25)
224  * @param group com.cosylab.vdct.graphics.objects.Group
225  * @param deep boolean
226  */
227 private void getInspectable(Group group, Vector objs, boolean deep) {
228 	Enumeration e = group.getSubObjectsV().elements();
229 	Object obj;
230 	while (e.hasMoreElements()) {
231 		obj = e.nextElement();
232 		if (obj instanceof com.cosylab.vdct.inspector.Inspectable)
233 			objs.addElement(obj);
234 		else if (deep && (obj instanceof Group))
235 			getInspectable((Group)obj, objs, deep);
236 	}
237 }
