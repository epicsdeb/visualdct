3   import java.awt.Dimension;
4   import java.util.Vector;
5   
6   import javax.swing.JTree;
7   import javax.swing.event.TreeSelectionEvent;
8   import javax.swing.event.TreeSelectionListener;
9   import javax.swing.tree.TreePath;
10  import javax.swing.tree.TreeSelectionModel;
11  
12  
13  // Tree Display for IOCs and Epics Groups
14  public class RdbDataTree extends JTree implements TreeSelectionListener {
15      
16  	public static final boolean debug=false;
17      private Vector listeners;
18      
19      public RdbDataTree() {
20          listeners = new Vector();
21          
22          getSelectionModel().setSelectionMode(TreeSelectionModel.SINGLE_TREE_SELECTION);
23          addTreeSelectionListener(this);
24          setPreferredSize(new Dimension(500, 500));
25      }
26  
27      public void addListener(RdbDataTreeListener l) {
28      	listeners.add(l);
29      }
30  
31      public void removeListener(RdbDataTreeListener l) {
32      	listeners.remove(l);
33      }
34  
35      // interface TreeSelectionListener
36      public void valueChanged(TreeSelectionEvent event) {
37      	
38      	// Idea: Path = {root, ioc, group, version}
39          TreePath path = event.getNewLeadSelectionPath();
40          if (path == null) { 
41              return;
42          }    
43          int pathLen = path.getPathCount();
44          if (pathLen < 2) { 
45          	// only root selected
46              return;
47          }
48  
49          String iocId = path.getPathComponent(1).toString();
50          
51          if (debug) {
52              System.out.println("Selected IOC  : " + iocId);
53          }
54          
55          if (pathLen == 2) {
56          	// only ioc selected
57              for (int i = 0; i < listeners.size(); i++) {
58                  ((RdbDataTreeListener)listeners.get(i)).iocSelected(iocId);
59              }
60              return;
61          }
62          
63          String groupId = path.getPathComponent(2).toString();
64          if (debug) {
65              System.out.println("Selected Group: " + groupId);
66          }
67  
68          if (pathLen == 3) {
69              for (int i = 0; i < listeners.size(); i++) {
70                  ((RdbDataTreeListener)listeners.get(i)).groupSelected(iocId, groupId);
71              }
72              return;
73          }
74          
75          String version = path.getPathComponent(3).toString();
76          if (debug) {
77              System.out.println("Selected version: " + version);
78          }
79  
80          if (pathLen == 4) {
81              for (int i = 0; i < listeners.size(); i++) {
82                  ((RdbDataTreeListener)listeners.get(i)).versionSelected(iocId, groupId, version);
83              }
84          }
85      }
86  };
