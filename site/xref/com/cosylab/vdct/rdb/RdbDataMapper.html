31  import java.sql.Connection;
32  import java.sql.ResultSet;
33  import java.sql.SQLException;

42  public class RdbDataMapper {
43  
44  	private RdbConnection helper = null;
45  
46  	public RdbDataMapper() throws Exception {
47  		helper = new RdbConnection();
48  	}
49  	
50  	public void setConnectionParameters(String host, String database, String user, String password) {
51  		helper.setParameters(host, database, user, password);
52  	}
53  
54  	public DBData loadRdbData(Object dsId, RdbDataId dataId) throws Exception {
55  
56  		DBData data = null;
57  		Exception exception = null;
58  		try {
59  			RdbDbContext context = new RdbDbContext(dsId, dataId, helper);
60  			data = context.load();
61  		} catch (SQLException sqlException) {
62  			sqlException.printStackTrace();
63  			exception = new Exception("Error while loading database: " + sqlException.getMessage());
64  		} catch (IllegalArgumentException illegalArgumentException) {
65  		    // Nothing
66  		}
67  		
68  		if (exception != null) {
69  			throw exception;
70  		}
71  		return data;
72  	}
73  	
74  	public boolean saveRdbData(Object dsId, RdbDataId dataId) throws Exception {
75  		
76  		boolean success = false;
77  		Exception exception = null;
78  		try {
79  			RdbDbContext context = new RdbDbContext(dsId, dataId, helper);
80  			context.save();
81  			helper.commit();
82  
83  			success = true;
84  		} catch (SQLException sqlException) {
85  			sqlException.printStackTrace();
86  			exception = new Exception("Error while saving database: " + sqlException.getMessage());
87  			helper.rollbackConnection();
88  		} catch (IllegalArgumentException illegalArgumentException) {
89  		    // Nothing
90  		}
91  			
92  		if (exception != null) {
93  			throw exception;
94  		}
95  		return success;
96  	}
97  	
98  	public Connection createNewConnection() throws SQLException {
99  		return helper.createConnection();
100 	}
101 
102 	public boolean isConnection() {
103 		return helper.isConnection();
104 	}
105 	
106 	public void closeConnection() throws SQLException {
107 		helper.closeConnection();
108 	}
109 	
110 	public int createAnIoc() throws SQLException {
111 		int iocId = saveIoc();
112         helper.commit();
113 		return iocId;
114 	}
115 	
116 	/** Returns Vector of String objects representing IOCs.
117      * @return 
118      * @throws java.sql.SQLException
119 	 */ 
120 	public Vector getIocs() throws SQLException {
121 
122 		Object[] columns = {"ioc_id"};  
123 		Object[][] conditions = {{}, {}};
124 		ResultSet set = helper.loadRows("ioc", columns, conditions, null, "ioc_id");
125         
126         Vector iocs = new Vector();
127 		while (set.next()) {
128         	iocs.add(set.getString(1));
129         }
130         return iocs;
131 	}
132 
133 	/** Returns Vector of String objects representing db files under the given IOC.
134      * @param iocId
135      * @return 
136      * @throws java.sql.SQLException
137 	 */ 
138 	public Vector getRdbDatas(String iocId) throws SQLException {
139 
140 		Object[] columns = {"p_db_file_name"};  
141 		Object[][] conditions = {{"ioc_id_FK"}, {iocId}};
142 		ResultSet set = helper.loadRows("p_db", columns, conditions, null, "p_db_file_name");
143         
144         Vector groups = new Vector();
145         while (set.next()) {
146         	groups.add(set.getString(1));
147         }
148         return groups;
149 	}
150 
151 	/** Returns Vector of String objects representing versions of the given group.
152      * @param group
153      * @param iocId
154      * @return 
155      * @throws java.sql.SQLException 
156 	 */ 
157 	public Vector getVersions(String group, String iocId) throws SQLException {
158 
159 		Object[] columns = {"p_db_version"};  
160 		Object[][] conditions = {{"ioc_id_FK", "p_db_file_name"}, {iocId, group}};
161 		ResultSet set = helper.loadRows("p_db", columns, conditions, null, "p_db_version");
162         
163         Vector versions = new Vector();
164         while (set.next()) {
165         	versions.add(set.getString(1));
166         }
167         return versions;
168 	}

170 	public void addRdbDataId(RdbDataId dataId, String desription) throws SQLException {
171 
172     	Object[][] keyPairs = {{"p_db_file_name", "ioc_id_FK", "p_db_version"},
173     			{dataId.getFileName(), dataId.getIoc(), dataId.getVersion()}};
174 		Object[][] valuePairs = {{"p_db_desc"}, {desription}};
175    		helper.saveRow("p_db", keyPairs, valuePairs);
176         helper.commit();
177 	}
178 	
179 	private int saveIoc() throws SQLException {
180 		Object[][] keyPairs = {{}, {}};
181 		Object[][] valuePairs = {{}, {}};
182 		helper.appendRow("ioc", keyPairs, valuePairs);
183 		
184 		Object[] columns = {"ioc_id"};
185 		ResultSet set = helper.loadRows("ioc", columns, keyPairs);
186 		if (!set.next()) {
187 			throw new SQLException("Could not create new ioc.");
188 		}
189 		return set.getInt(1);
190 	}
