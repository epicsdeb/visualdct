131 	private String getList(Object[] elements) {
132 		StringBuffer buffer = new StringBuffer();
133 		int len = elements.length;
134 		for (int e = 0; e < len; e++) {
135 			buffer.append(elements[e].toString());
136 			if (e < len - 1) {
137 				buffer.append(",");
138 			}
139 		}
140 		return buffer.toString();
141 	}
142 
143 	private String getQuestionMarkList(int length) {
144 		StringBuffer buffer = new StringBuffer();
145 		for (int e = 0; e < length; e++) {
146 			buffer.append("?");
147 			if (e < length - 1) {
148 				buffer.append(",");
149 			}
150 		}
151 		return buffer.toString();
152 	}
153 	
154 	private String getEqualityList(Object[] elements) {
155 		return getEqualityExpression(elements, ",");
156 	}
157 
158 	private String getEqualityStatement(Object[] elements) {
159 		return getEqualityExpression(elements, " AND ");
160 	}
161 
162 	private String getEqualityExpression(Object[] elements, String delimiter) {
163 		StringBuffer buffer = new StringBuffer();
164 		int len = elements.length;
165 		for (int e = 0; e < len; e++) {
166 			buffer.append(elements[e].toString());
167 			buffer.append("=?");
168 			if (e < len - 1) {
169 				buffer.append(delimiter);
170 			}
171 		}
172 		return buffer.toString();
173 	}

175 	private void insertValues(PreparedStatement statement, Object[] elements) throws SQLException {
176 		insertValues(statement, elements, 0);
177 	}
178 
179 	private void insertValues(PreparedStatement statement, Object[] elements1, Object[] elements2) throws SQLException {
180 		insertValues(statement, elements1, 0);
181 		insertValues(statement, elements2, elements1.length);
182 	}
183 	
184 	private void insertValues(PreparedStatement statement, Object[] elements, int offset) throws SQLException {
185 		for (int i = 0; i < elements.length; i++) {
186         	statement.setString(i + offset + 1, String.valueOf(elements[i]));
187         }
188 	}

215     /**
216      *
217      * @param table table
218      * @param columns columns
219      * @param keyPairs keyPairs
220      * @param conditions conditions
221      * @param orderBy orderBy
222      * @return something
223      * @throws SQLException foo
224      */
225     public ResultSet loadRows(String table, Object[] columns, Object[][] keyPairs, String conditions, String orderBy) throws SQLException {
226 
227         String columnsString = columns != null ? getList(columns) : "*";
228         String equalityConditions = getEqualityStatement(keyPairs[0]);
229         boolean equalityExists = !equalityConditions.equals(""); 
230         
231         String sqlString = "SELECT DISTINCT " + columnsString + " FROM " + table;
232         if (equalityExists) {
233             sqlString += " WHERE " + equalityConditions;
234         }
235         if (equalityExists && conditions != null) {
236             sqlString += " AND ";
237         }
238         if (conditions != null) {
239             sqlString += conditions;
240         }
241         if (orderBy != null) {
242         	sqlString += " ORDER BY " + orderBy;
243         }
244         PreparedStatement statement = connection.prepareStatement(sqlString);
245         insertValues(statement, keyPairs[1]);
246         selectCount++;
247         return statement.executeQuery();
248 	}

250 	private boolean isRowPresent(String table, Object[][] keyPairs) throws SQLException {
251 		ResultSet set = loadRows(table, null, keyPairs);
252 		return set.next();
253 	}

257     /**
258      *
259      * @param table table
260      * @param keyPairs keyPairs
261      * @param valuePairs valuePairs
262      * @throws SQLException foo
263      */
264 	public void insertRow(String table, Object[][] keyPairs, Object[][] valuePairs) throws SQLException {
265 		int keyLen = keyPairs[1].length;
266 		int valueLen = valuePairs[1].length;
267 
268 		String columnString = getList(keyPairs[0]) + (valueLen > 0 ?  "," + getList(valuePairs[0]) : "");
269 		String valueString = getQuestionMarkList(keyLen + valueLen);
270 		String sqlString = "INSERT INTO " + table
271 		+ " (" + columnString + ") VALUES (" + valueString + ")";
272 
273 		PreparedStatement statement = connection.prepareStatement(sqlString);
274 		insertValues(statement, keyPairs[1], valuePairs[1]);
275 		statement.execute();
276 		insertCount++;
277 	}

283      * @param table table
284      * @param keyPairs keyPairs
285      * @param valuePairs valuePairs
286      * @throws SQLException foo
287      */
288 	public void updateRow(String table, Object[][] keyPairs, Object[][] valuePairs) throws SQLException {
289 		if (valuePairs[0].length > 0) {
290 			String condition = getEqualityStatement(keyPairs[0]);
291 			String setString = getEqualityList(valuePairs[0]);
292 			String sqlString = "UPDATE " + table + " SET " + setString + " WHERE " + condition;
293 
294 			PreparedStatement statement = connection.prepareStatement(sqlString);
295 			insertValues(statement, valuePairs[1], keyPairs[1]);
296 			statement.execute();
297 			updateCount++;
298 		}
299 	}

307      * @throws SQLException foo

309 	public void deleteRows(String table, Object[][] keyPairs) throws SQLException {
310 		String sqlString = "DELETE FROM " + table + " WHERE " + getEqualityStatement(keyPairs[0]);
311 		PreparedStatement statement = connection.prepareStatement(sqlString);
312         insertValues(statement, keyPairs[1]);
313 
314         statement.execute();
315         deleteCount++;
316 	}

324      * @param valuePairs valuePairs

327 	public void saveRow(String table, Object[][] keyPairs, Object[][] valuePairs) throws SQLException {
328 		if (!isRowPresent(table, keyPairs)) {
329 			insertRow(table, keyPairs, valuePairs);
330 		} else {
331 			updateRow(table, keyPairs, valuePairs);
332 		}
333 	}
