320 	private int validateFont(double scale, int rwidth, int height) {
321 	    
322 	    int irheight = (int)(scale*height);
323 
324 		  // set appropriate font size
325 		  int x0 = (int)(24*scale);		// insets
326 		  int y0 = (int)(12*scale);
327 
328 
329 		  // fields
330 		  int fieldRows = Math.max(leftFields, rightFields);
331 		  height += fieldRows * Constants.FIELD_HEIGHT;
332 		 // height = Math.max(height, Constants.TEMPLATE_MIN_HEIGHT);
333 		  int frheight = (int)(scale*height);
334 
335 //		  rlinkY = frheight;
336 		  
337 		  // properties
338 
339 		  int xx0 = (int)(14*scale);		// insets
340 		  int yy0 = (int)(8*scale);
341 
342 
343 		 // !!! optimize - static
344 
345 		  rfieldRowHeight = (irheight-2*y0)*0.175;
346 
347 		  if (rwidth<(2*xx0)) fieldFont = null;
348 		  else
349 			  fieldFont = FontMetricsBuffer.getInstance().getAppropriateFont(
350 				  			 Constants.DEFAULT_FONT, Font.PLAIN, 
351 			 	 			 fieldMaxStr, rwidth-x0, (int)rfieldRowHeight);
352 
353 		  int ascent = 0;
354 		  //rfieldRowHeight = 0;
355 		  if (fieldFont!=null)
356 		  {
357 			  FontMetrics fm = FontMetricsBuffer.getInstance().getFontMetrics(fieldFont);
358 			  rfieldLabelX = xx0;
359 		 	  rfieldLabelY = frheight+2*fm.getAscent();
360 			  //rfieldRowHeight = fm.getHeight();
361 			  ascent = fm.getAscent();
362 		  }
363 
364 		  int rheight = frheight + yy0 + (int)(rfieldRowHeight*templateData.getProperties().size())+ascent;
365 		  
366 		
367 		  // description
368 
369 		  int idLabelHeight = (int)(Constants.FIELD_HEIGHT*scale);
370 		  
371 		  setLabel(getDescription());
372 		
373 		  final int MAX_FONT_SIZE = 30;
374 		  Font font;
375 		  font = FontMetricsBuffer.getInstance().getAppropriateFont(
376 			  			Constants.DEFAULT_FONT, Font.PLAIN, 
377 			  			getLabel(), rwidth-x0, irheight-y0/*-idLabelHeight*/, (int)(MAX_FONT_SIZE*scale));
378 		  
379 		  if (rwidth<(2*x0)) font = null;
380 		  else
381 		  if (font!=null) {
382 			  FontMetrics fm = FontMetricsBuffer.getInstance().getFontMetrics(font);
383 			  setRlabelX((rwidth-fm.stringWidth(getLabel()))/2);
384 		 	  setRlabelY((irheight-fm.getHeight()/*+idLabelHeight*/)/2+fm.getAscent());
385 		  }
386 		  setFont(font);
387 
388 		  // id label
389 		  
390 		  // idlabel = templateData.getTemplate().getId();
391 		  idlabel = templateData.getName();
392 		  if (rwidth<(2*x0)) idFont = null;
393 		  else
394 			  idFont = FontMetricsBuffer.getInstance().getAppropriateFont(
395 				  			 Constants.DEFAULT_FONT, Font.PLAIN, 
396 			 	 			 idlabel, rwidth-x0, idLabelHeight);
397 		
398 		  if (idFont!=null) {
399 			  FontMetrics fm = FontMetricsBuffer.getInstance().getFontMetrics(idFont);
400 			  ridLabelX = (rwidth-fm.stringWidth(idlabel))/2;
401 		 	  ridLabelY = (idLabelHeight-fm.getHeight())/2+fm.getAscent();
402 		  }
403 		  
404 		  return rheight;
405 	}

408 	 * @see com.cosylab.vdct.graphics.objects.VisibleObject#validate()
409 	 */
410 	private boolean validating = false;
411 	protected void validate()
412 	{
413 		if (validating)
414 			return;
415 		
416 		try {
417 			validating = true;
418 
419 			  // template change check
420 			  VDBTemplate tmpl = DataSynchronizer.getInstance().getTemplate(getDsId(), getTemplateData().getTemplate().getId());		
421 			  if (tmpl!=getTemplateData().getTemplate())
422 			  {
423 			  	getTemplateData().setTemplate(tmpl);
424 				synchronizeLinkFields();
425 			  }
426 			  else
427 			  {
428 			  	if (getTemplateData().getTemplate().getPortsGeneratedID()!=portsID)
429 					synchronizePortLinkFields();
430 				if (getTemplateData().getTemplate().getMacrosGeneratedID()!=macrosID)
431 					synchronizeMacroLinkFields();
432 			  }
433 			
434 			  revalidatePosition();
435 
436 			  double scale = getRscale();
437 			  int rwidth = (int)((getX()+getWidth())*scale)-getRx();
438 			  int height = (int)(getY()*scale+Constants.TEMPLATE_INITIAL_HEIGHT)-getRy();
439 
440 			  initY = height;
441 			  	  
442 			  int rheight = validateFont(scale, rwidth, height);
443 			  
444 			  setHeight((int)(rheight/scale));
445 			  setRwidth(rwidth);
446 			  setRheight(rheight);
447 			  
448 		      // sub-components
449 			  revalidatePosition();
450 			  validateFields();
451 			  
452 		}
453 		finally {
454 			validating = false;
455 		}
456 	}
