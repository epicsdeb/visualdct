254 public void destroy() {
255 	if (!isDestroyed()) {
256 		super.destroy();
257 
258 		if (outlinks.size()>0) {
259 			Object[] objs = new Object[outlinks.size()];
260 			outlinks.copyInto(objs);
261 			for(int i=0; i<objs.length; i++) {
262 				OutLink outlink = (OutLink)objs[i];
263 				OutLink start = EPICSLinkOut.getStartPoint(outlink);
264 				if((start instanceof EPICSLinkOut)){
265 					((EPICSLinkOut)start).sourceDestroyed();
266 				}
267 				else if ((start instanceof EPICSVarLink)) {
268 				    
269 				    if (outlink instanceof Connector) {
270 				        OutLink temp; 
271 				        while(outlink instanceof Connector) {
272 					        temp =((Connector)outlink).getOutput();
273 					        ((Connector)outlink).destroy();
274 					        outlink = temp;
275 					    }
276 				    }
277 				    start.disconnect(this);
278 				    
279 				}
280 				else if (start!=null)
281 					start.disconnect(this);
282 				else 
283 					outlink.disconnect(this);
284 			}
285 			outlinks.clear();
286 		}
287 		
288 		if (lastUpdatedFullName!=null)
289 			((Group)getRootContainer()).getLookupTable().remove(data.getFullName());
290 		//else
291 		//	((LinkManagerObject)getParent()).removeInvalidLink(this);
292 
293 		data.setVisibleObject(null);
294 		getParent().removeObject(getName());
295 
296 		// also remove macro definition
297 		data.getTemplate().removeMacro(getName());
298 		
299 
300 		
301 	}
302 	
303 }

309 public void disconnect(Linkable disconnector) {

312 		
313 		if (outlinks.size()==0) {
314 		 	// cannot be destoryed by removing links
315 			//destroy();
316 			disconnected = true;
317 		}
318 		else
319 		 if (outlinks.size()==1)
320 			if (outlinks.firstElement() instanceof VisibleObject)
321 				setColor(((VisibleObject)outlinks.firstElement()).getColor());
322 	}

329  */
330 protected void draw(java.awt.Graphics g, boolean hilited) {
331 
332 	ViewState view = ViewState.getInstance(getDsId());
333 	
334 	double Rscale = getRscale();
335 	boolean zoom = Rscale < 1.0 && view.isZoomOnHilited() && view.isHilitedObject(this);
336 	if (zoom) {
337 	    zoomImage = ZoomPane.getInstance().startZooming(this, true);
338 	}
339 
340 	int rrx = getRx() - view.getRx();
341 	int rry = getRy() - view.getRy();
342 
343 	int rwidth = getRwidth();
344 	int rheight = getRheight();
345 
346 	boolean rightSide = isRight();
347 
348 	// clipping
349 	if (!((rrx > view.getViewWidth())
350 		|| (rry > view.getViewHeight())
351 		|| ((rrx + rwidth) < 0)
352 		|| ((rry + rheight) < 0)) || isZoomRepaint()) {
353 	    
354 	    if (isZoomRepaint()) {
355 	        rrx = ZoomPane.getInstance().getLeftOffset();
356 	        rry = ZoomPane.getInstance().getTopOffset();
357 	    }
358 	    
359 	    Polygon poly = null;
360 		if (rightSide)
361 		{
362 			poly = rightPoly;
363 			poly.translate(rrx-rightXtranslation, rry-rightYtranslation);
364 			rightXtranslation = rrx; rightYtranslation = rry;
365 		}
366 		else
367 		{
368 			poly = leftPoly;
369 			poly.translate(rrx-leftXtranslation, rry-leftYtranslation);
370 			leftXtranslation = rrx; leftYtranslation = rry;
371 		}
372 		
373 	
374 		if (!hilited)
375 			g.setColor(Constants.RECORD_COLOR);
376 		else
377 			if (view.isPicked(this))
378 				g.setColor(Constants.PICK_COLOR);
379 			else
380 				if (view.isSelected(this) || view.isBlinking(this))
381 					g.setColor(Constants.SELECTION_COLOR);
382 				else
383 					g.setColor(Constants.RECORD_COLOR);
384 	
385 		g.fillPolygon(poly);
386 
387 		Color drawColor;
388 		if (!hilited)
389 			drawColor = Constants.FRAME_COLOR;
390 		else
391 			drawColor =
392 				(view.isHilitedObject(this))
393 					? Constants.HILITE_COLOR
394 					: Constants.FRAME_COLOR;
395 
396 		g.setColor(drawColor);
397 		g.drawPolygon(poly);
398 		
399 		if (rightSide)
400 		{
401 			int xx = rrx-r;
402 			g.drawOval(xx, rry+(rheight-r)/2, r, r);
403 
404 			// draw tail if there are any links
405 			if (!disconnected)
406 			{
407 				int yy = rry+rheight/2;
408 				g.drawLine(xx, yy, xx-r, yy);
409 			}
410 		}
411 		else
412 		{
413 			int xx = rrx+rwidth;
414 			g.drawOval(xx, rry+(rheight-r)/2, r, r);
415 
416 			// draw tail if there are any links
417 			if (!disconnected)
418 			{
419 				int yy = rry+rheight/2;
420 				xx += r;
421 				g.drawLine(xx, yy, xx+r, yy);
422 			}
423 		}
424 
425 		Font font = FontMetricsBuffer.getInstance().getAppropriateFont(
426 	  			Constants.DEFAULT_FONT, Font.PLAIN, 
427 	  			getLabel(), rwidth*4, rheight);
428 
429 		if (font!=null)	{
430 		    FontMetrics fm = FontMetricsBuffer.getInstance().getFontMetrics(font);
431 		    if (textPositionNorth) {
432 		        setRlabelX(rwidth/2-fm.stringWidth(getLabel())/2);
433 		        setRlabelY(-fm.getHeight()+fm.getAscent());
434 		    } else {
435 		        int s = (int)(getScale()*Constants.LINK_STUB_SIZE/4.0);
436 		        setRlabelY((rheight + fm.getAscent() - fm.getDescent())/2);
437 		        if (isRight()) {
438 		            setRlabelX(rwidth + s);
439 		        } else {
440 		            setRlabelX(-fm.stringWidth(getLabel())-s);
441 		        }
442 		    }
443 		    g.setFont(getFont());
444 			g.drawString(getLabel(), rrx+getRlabelX(), rry+getRlabelY());
445 		}		
446 		
447 	}
448 
449 	if (false)///!!!
450 	{
451 		
452 		// draw link
453 		g.setColor(getVisibleColor());
454 
455 		//LinkDrawer.drawLink(g, this, inlink, getQueueCount(), rightSide);
456 	}
457 	
458 	if (zoom) {
459         rwidth /= Rscale;
460         rheight /= Rscale;
461         rrx -= ((rwidth - getRwidth())/2 + ZoomPane.getInstance().getLeftOffset());
462         rry -= ((rheight - getRheight())/2 + ZoomPane.getInstance().getTopOffset());
463         if (view.getRx() < 0)
464             rrx = rrx < 0 ? 2 : rrx;
465         if (view.getRy() < 0) 
466             rry = rry <= 0 ? 2 : rry;
467         g.drawImage(zoomImage, rrx,rry, ZoomPane.getInstance());
468     }
469 
470 		
471 }

665 private void validateFontAndPolygon(double Rscale, int rwidth, int rheight) {
666     
667     leftXtranslation = 0; leftYtranslation = 0;
668     rightXtranslation = 0; rightYtranslation = 0;
669     
670     if (getMode() == InLink.OUTPUT_MACRO_MODE)
671     {
672       setWidth(Constants.GRID_SIZE);
673   	  setHeight(Constants.GRID_SIZE);
674 
675   	  // left poly
676   	  leftPoly.xpoints[0]=0;
677   	  leftPoly.xpoints[1]=rwidth/2;
678   	  leftPoly.xpoints[2]=rwidth;
679   	  leftPoly.xpoints[3]=leftPoly.xpoints[2];
680   	  leftPoly.xpoints[4]=leftPoly.xpoints[1];
681   	
682   	  leftPoly.ypoints[0]=rheight/2;
683   	  leftPoly.ypoints[1]=rheight;
684   	  leftPoly.ypoints[2]=leftPoly.ypoints[1];
685   	  leftPoly.ypoints[3]=0;
686   	  leftPoly.ypoints[4]=leftPoly.ypoints[3];
687 
688
