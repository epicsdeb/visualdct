101 	/**
102 	 * Recursively resolves macros/ports
103 	 * @param name
104 	 * @param value
105 	 * @return
106 	 */
107 	public String resolveMacro(String name, String value) {
108 		//DebugSystem.out.println("resolve macro "+name+"="+value);
109 					
110 		// only macros/ports on previous level can have influence	
111 		if (getParent()!=null) {
112 			/* try with a record first 
113 			 * if there are some macros in the target, we suppose the same macros
114 		 	 * are in the record name, therefore they are substituted later
115 			 */
116 			value = getParent().resolveLink(value);
117 			value = getParent().matchAndReplace(value);
118 		}
119 			
120 		return value;
121 	}

125 	/**
126 	 * Creates or returns NamingContext for specific template
127 	 * @param instance
128 	 * @return
129 	 */
130 	public NamingContext createNamingContextFor(VDBTemplateInstance instance) {
131 		//DebugSystem.out.println("create record namer for " +instance);
132 		
133 		NamingContext rn=(NamingContext)namingContextCache.get(instance);
134 		if (rn==null) {
135 			//handle added and removed prefixes
136 			String addedPrefix=this.addedPrefix;
137 			String removedPrefix=this.removedPrefix;
138 			if (Settings.getInstance().getHierarhicalNames() && export) { 
139 				addedPrefix=addedPrefix+instance.getName()+Constants.HIERARCHY_SEPARATOR;
140 			}
141 			
142 			rn=new NamingContext(this, instance, addedPrefix, removedPrefix, export);
143 			namingContextCache.put(instance, rn);
144 		}
145 		return rn;
146 	}

149 	 * Recursively resolves macros/ports
150 	 * @param port
151 	 * @return
152 	 */
153 	public String resolvePort(VDBPort port) {
154 		//DebugSystem.out.println("resolve port "+port);
155 		
156 		String target = port.getTarget();
157 
158 		/* try with a record first 
159 		 * if there are some macros in the target, we suppose the same macros
160 		 * are in the record name, therefore they are substituted later
161 		 */
162 		
163 		LinkProperties lp = new LinkProperties(port);								
164 		Object rec = template.getGroup().findObject(lp.getRecord(), true);
165 		if (rec !=null && rec instanceof Record) {
166 			/*lp.setRecord(getResolvedName(((Record)rec).getRecordData()));
167 			target = lp.getFull();*/
168 			target=getResolvedName(target);
169 		}
170 		
171 		target = matchAndReplace(target);
172 		
173 		return target;
174 	}

226 	 * Subtracts record name from link, looks for apropriate record and renames
227 	 * the record in apropriate way. It contructs back the link.
228 	 * @param target
229 	 * @return
230 	 */
231 	public String resolveLink(String target) {
232 		//DebugSystem.out.println("resolve link "+target);
233 		
234 		String record=LinkProperties.getRecordFromString(target);
235 		Object rec=null;								
236 		
237 		if (record!=null && template !=null) rec=template.getGroup().findObject(record, true);
238 		if (rec !=null && rec instanceof Record) {
239 			target=getResolvedName(target);
240 		}
241 		
242 		return target;
243 	}

247 	/**
248 	 * Searches for the macro in the specific template.
249 	 * Possibly also for global macro definitions.
250 	 * @param name
251 	 * @return
252 	 */
253 	public String findAndResolveMacro(String name) {
254 		//DebugSystem.out.println("find and resolve macro "+name);
255 		
256 		String value = "$("+name+")";
257 		
258 		if (macroMap.containsKey(name)) {
259 			if (macroMap.get(name) == cycleFlag) {
260 				Console.getInstance().println("Warning: cyclic reference made by macro '"+value+"'.");
261 				
262 				addMacro(name, value);
263 			}
264 			return (String)macroMap.get(name);
265 		}
266 		
267 		macroMap.put(name, cycleFlag); // flag visited
268 				
269 		if (templateInstance!=null) {
270 			String val = (String)templateInstance.getProperties().get(name);
271 			if (val!=null) value = resolveMacro(name, val);
272 			else {
273 				if (Settings.getInstance().getGlobalMacros() && getParent()!=null) 
274 							value = getParent().findAndResolveMacro(name);		
275 			}
276 		}
277 		
278 		
279 				
280 		addMacro(name, value);
281 				
282 		return value;
283 	}

285 	/**
286 	 * Searches for port in the specific template.
287 	 * Notice that port has to be resolved in inner template, but added in outer one.
288 	 * @param temp
289 	 * @param name
290 	 * @return
291 	 */
292 	public String findAndResolvePort(String temp, String name) {
293 		//DebugSystem.out.println("find and resolve port: "+temp+"."+name);
294 		
295 		String portname=temp+"."+name;
296 		String target="$("+portname+")";
297 		
298 		if (portMap.containsKey(portname)) {
299 			if (portMap.get(portname) == cycleFlag) {
300 				Console.getInstance().println("Warning: cyclic reference made by port '"+portname+"'.");
301 				addPort(portname, target);
302 			}
303 			return (String)portMap.get(portname);
304 		}
305 		
306 		portMap.put(portname, cycleFlag); // flag visited	
307 		
308 		
309 		
310 		Object obj=template.getGroup().findObject(temp, true);
311 		if (obj!=null && obj instanceof Template) {
312 			Template t=(Template)obj;
313 			NamingContext newNamer = createNamingContextFor(t.getTemplateData());
314 
315 			VDBPort port=(VDBPort)t.getTemplateData().getTemplate().getPorts().get(name);
316 			if (port!=null) target=newNamer.resolvePort(port);		
317 		}
318 		
319 		addPort(portname, target);
320 		
321 		return target;
322 	}

324 	/**
325 	 * Finds $() and $(.) in strings and replaces them.
326 	 * TASK:RECMACROS: currently it doesn't handle something like $($()) very well.
327 	 * I don't think this is a simple problem, which could be done with regex.
328 	 * Hierarchy prevents it. 
329 	 * @param value
330 	 * @return
331 	 */
332 	public String matchAndReplace(String value) {
333 		if (value==null || value.indexOf('$')<0) return value;
334 		
335 		//DebugSystem.out.println("matchandreplace "+value);
336 		
337 		// by definition from ebnfg - \0 is bad character, but occurs in groups
338 		//Pattern macrop = Pattern.compile("$\\(([a-zA-Z0-9_:-]+)\\)");
339 		//Pattern portp = Pattern.compile("$\\(([a-zA-Z0-9_:-]+)\\.([a-zA-Z0-9_:-]+)\\)");
340 		Pattern macrop = Pattern.compile("\\$\\(([^\\.\\$\\)]+)\\)");
341 		//Pattern portp = Pattern.compile("\\$\\(([^\\.\\$]+)\\.([^\\.\\$]+)\\)");
342 		Pattern portp = Pattern.compile("\\$\\(([^\\.]+)\\.([^\\.]+)\\)");
343 		
344 		Matcher port = portp.matcher(value);
345 		StringBuffer result = new StringBuffer();
346 		while (port.find()) {
347 			String portt=port.group(1), portn=port.group(2);
348 			port.appendReplacement(result, findAndResolvePort(portt, portn).replaceAll("\\$","\\\\\\$"));
349 		}
350 		port.appendTail(result);
351 		
352 		Matcher macro = macrop.matcher(result.toString());
353 		result=new StringBuffer();
354 		while (macro.find()) {
355 			String macron=macro.group(1);
356 			macro.appendReplacement(result, findAndResolveMacro(macron).replaceAll("\\$","\\\\\\$"));
357 		}
358 		macro.appendTail(result);
359 		
360 		return result.toString();
361 	}
