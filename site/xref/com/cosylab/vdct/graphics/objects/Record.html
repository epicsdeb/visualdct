578 /**
579  * Draws a timeout clock.
580  * @param g g
581  * @param hour hour
582  * @param minute minute
583  * @param x0 x0
584  * @param y0 y0
585  * @param w w
586  * @param h h
587  */
588 protected static void drawDebugTimeout(Graphics g, int hour, int minute, int x0, int y0, int w, int h)
589 {
590 	g.setColor(Color.MAGENTA);
591 	
592 	final int thickness = 3;
593 	if (w < 5*thickness || h < 5*thickness) return;
594 
595 	// calculate angles	
596 	double phiHour = (2*Math.PI/12)*hour-Math.PI/2;
597 	double phiMinute = (2*Math.PI/60)*minute-Math.PI/2;
598 
599 	int size = (int)Math.min(0.75 * h, 0.75 * w);
600 	
601 	// optimize
602 	w = w/2;
603 	h = h/2;
604 	size = size/2;
605 	double cosHour = Math.cos(phiHour);
606 	double sinHour = Math.sin(phiHour);
607 	double cosMin = Math.cos(phiMinute);
608 	double sinMin = Math.sin(phiMinute);
609 	
610 	g.drawOval(x0 + w - size + thickness, y0 + h - size + thickness, 2*size - 2*thickness, 2*size - 2*thickness);
611 	g.drawOval(x0 + w - size, y0 + h - size, size*2, size*2);
612 	
613 	int pointerSize = (3 * size) / 4;
614 	g.drawLine(x0 + w,     y0 + h, x0 + (int)(w + pointerSize * cosMin),      y0 + (int)(h + pointerSize * sinMin));
615 	g.drawLine(x0 + w + 1, y0 + h, x0 + (int)(w + pointerSize * cosMin) + 1,  y0 + (int)(h + pointerSize * sinMin));
616 	
617 	pointerSize = size / 2;
618 	g.drawLine(x0 + w,     y0 + h, x0 + (int)(w + pointerSize * cosHour),      y0 + (int)(h + pointerSize * sinHour));
619 	g.drawLine(x0 + w + 1, y0 + h, x0 + (int)(w + pointerSize * cosHour) + 1,  y0 + (int)(h + pointerSize * sinHour));
620 
621 	int[] xs2 = { x0 + w - 5, x0 + w - 5, x0 + w + 5, x0 + w + 5};
622 	int[] ys2 = { y0 + h - size, y0 + h - size - 5, y0 + h - size - 5, y0 + h - size};
