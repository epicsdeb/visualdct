53  public void actionPerformed(ActionEvent event)
54  {
55  	String action = event.getActionCommand();
56  	if(action.equals(colorString))
57  	{
58  		Color newColor = ColorChooser.getColor(colorTitleString, getColor());
59  		
60  		if(newColor != null)
61  		{
62  			setColor(newColor);
63  			currentColor = newColor;
64  		}
65  		
66  		CommandManager.getInstance().execute("RepaintWorkspace");
67  	}
68  	else if(action.equals(dashedString))
69  	{
70  		dashed = !dashed;
71  		currentIsDashed = dashed;
72  
73  		CommandManager.getInstance().execute("RepaintWorkspace");
74  	}
75  }

96  private String getAvailableHashId()
97  {
98  	int number = 0;
99  	String testHashId = hashIdPrefix + String.valueOf(number);
100 	
101 	while(getParent().containsObject(testHashId))
102 	{
103 		number++;
104 		testHashId = hashIdPrefix + String.valueOf(number);		
105 	}
106 
107 	return testHashId;
108 }

127

130 	setHeight(Constants.CONNECTOR_HEIGHT);
131 	
132 	setColor(currentColor);
133 	dashed = currentIsDashed;
134 	
135 	if(parName == null)
136 	{
137 		hashId = getAvailableHashId();
138 	
139 		if(parentGroup.getAbsoluteName().length() > 0)
140 			name = parentGroup.getAbsoluteName() + Constants.GROUP_SEPARATOR + hashId;
141 		else
142 			name = hashId;
143 	}
144 	else
145 		name = parName;	
146 }

161 public Flexible copyToGroup(Object dsId, String group)
162 {
163 
164 	String newName;
165 	if(group.equals(nullString))
166 		newName = Group.substractObjectName(getName());
167 	else
168 		newName = group + Constants.GROUP_SEPARATOR + Group.substractObjectName(getName());
169 
170 // object with new name already exists, add suffix ///!!!
171 	while(Group.getRoot(dsId).findObject(newName, true) != null)
172 		newName = StringUtils.incrementName(newName, Constants.COPY_SUFFIX);
173 
174 	Box grBox = new Box(newName, Group.getRoot(dsId),
175 						startVertex.getX(), startVertex.getY(),
176 						endVertex.getX(), endVertex.getY());
177 	grBox.setColor(getColor());
178 	grBox.setParent(null);
179 	Group.getRoot(dsId).addSubObject(newName, grBox, true);

181 	//ViewState view = ViewState.getInstance();

184 	unconditionalValidation();

201 protected void draw(Graphics g, boolean hilited)
202 {
203 	ViewState view = ViewState.getInstance(getDsId());
204 	int offsetX = view.getRx();
205 	int offsetY = view.getRy();
206 	
207 	int posX = getRx() - offsetX;
208 	int posY = getRy() - offsetY;
209 	int rwidth = getRwidth();
210 	int rheight = getRheight();
211 
212 	double Rscale = view.getScale();
213 	boolean zoom = (Rscale < 1.0) && view.isZoomOnHilited() && view.isHilitedObject(this);
214 	if (zoom) {
215         rwidth /= Rscale;
216         rheight /= Rscale;
217         posX -= (rwidth - getRwidth())/2;
218         posY -= (rheight - getRheight())/2;
219         if (view.getRx() < 0)
220             posX = posX < 0 ? 2 : posX;
221         if (view.getRy() < 0) 
222             posY = posY <= 0 ? 2 : posY;
223         Rscale = 1.0;
224     }
225 	
226 	if (hilited)
227 		g.setColor(Constants.HILITE_COLOR);
228 	else
229 		g.setColor(getVisibleColor());
230 
231 	//double scale = view.getScale();
232 	
233 	posX = startVertex.getRx() - offsetX;
234 	posY = startVertex.getRy() - offsetY;
235 		
236 	int posX2 = endVertex.getRx() - offsetX;
237 	int posY2 = endVertex.getRy() - offsetY;
238 
239 	int t;
240 	if(posX > posX2)
241 	{
242 		t = posX;
243 		posX = posX2;
244 		posX2 = t;
245 	}
246 	if(posY > posY2)
247 	{
248 		t = posY;
249 		posY = posY2;
250 		posY2 = t;
251 	}
252 		
253 	if((dashed) && ((posX != posX2) || (posY != posY2)))
254 	{
255 		int curX = posX;
256 		while(curX <= posX2)
257 		{
258 			int curX2 = curX + Constants.DASHED_LINE_DENSITY;
259 				
260 			if(curX2 > posX2)
261 				curX2 = posX2;
262 				
263 			g.drawLine(curX, posY, curX2, posY);
264 			g.drawLine(curX, posY2, curX2, posY2);
265 				
266 			curX += 2 * Constants.DASHED_LINE_DENSITY;
267 		}
268 
269 		int curY = posY;
270 		while(curY <= posY2)
271 		{
272 			int curY2 = curY + Constants.DASHED_LINE_DENSITY;
273 				
274 			if(curY2 > posY2)
275 				curY2 = posY2;
276 				
277 			g.drawLine(posX, curY, posX, curY2);
278 			g.drawLine(posX2, curY, posX2, curY2);
279 				
280 			curY += 2 * Constants.DASHED_LINE_DENSITY;
281 		}
282 	}
283 	else
284 		g.drawRect(posX, posY, posX2 - posX, posY2 - posY);
285 }

350 public boolean moveToGroup(Object dsId, String group)
351 {
352 	String currentParent = Group.substractParentName(getName());
353 	if(group.equals(currentParent) && dsId.equals(getDsId()))
354 		return false;
355 	
356 	//String oldName = getName();
357 	String newName;
358 	if (group.equals(nullString))
359 		newName = Group.substractObjectName(getName());
360 	else
361 		newName = group + Constants.GROUP_SEPARATOR + Group.substractObjectName(getName());
362 
363 	// object with new name already exists, add suffix // !!!
364 	Object obj;
365 	boolean renameNeeded = false;
366 	while ((obj=Group.getRoot(dsId).findObject(newName, true))!=null)
367 	{
368 		if (obj==this)	// it's me :) already moved, fix data
369 		{
370 			name = newName;
371 			return true;
372 		}
373 		else
374 		{
375 			renameNeeded = true;
376 			newName = StringUtils.incrementName(newName, Constants.MOVE_SUFFIX);
377 		}
378 	}
379 
380 	if (renameNeeded)
381 		return rename(dsId, newName);
382 	
383 	getParent().removeObject(Group.substractObjectName(getName()));
384 	setParent(null);
385 	Group.getRoot(dsId).addSubObject(newName, this, true);
386 
387 	name = newName;
388 	unconditionalValidation();
389 
390 	return true;
391 }

393 public boolean rename(Object dsId, String newName)
394 {
395 	String newObjName = Group.substractObjectName(newName);
396 	String oldObjName = Group.substractObjectName(getName());
397 
398 	if(!oldObjName.equals(newObjName))
399 	{
400 		getParent().removeObject(oldObjName);
401 		String fullName = StringUtils.replaceEnding(getName(), oldObjName, newObjName);
402 		name = fullName;
403 		getParent().addSubObject(newObjName, this);
404 	}
405 	
406 // move if needed
407 	moveToGroup(dsId, Group.substractParentName(newName));
408 
409 	return true;
410 }

458 /**
459  * Creation date: (19.12.2000 20:20:20)
460  * @return com.cosylab.visible.objects.VisibleObject
461  * @param px int
462  * @param py int
463  */
464 public VisibleObject intersects(int px, int py) {
465 
466 	// first check on small sub-objects like connectors
467 	VisibleObject spotted = hiliteComponentsCheck(px, py);
468   	if (spotted == null) {
469 
470   		final int DIST = 5;
471   		
472   		int rx = getRx();
473   		int ry = getRy();
474   		int rw = getRwidth();
475   		int rh = getRheight();
476   		boolean insideOuter = ((rx-DIST)<=px) &&
477 							  ((ry-DIST)<=py) && 
478 							  ((rx+rw+DIST)>=px) && 
479 							  ((ry+rh+DIST)>=py);
480   		if (insideOuter)
481   		{
482  	 		boolean outsideInner = !(((rx+DIST)<px) &&
483 									((ry+DIST)<py) && 
484 									((rx+rw-DIST)>px) && 
485 									((ry+rh-DIST)>py));
486 	 		
487 	 		if (outsideInner)
488 	 			spotted = this;
489   		}
490   		
491   	}
492   	
493   	return spotted;
494 }
