32  /**
33   * @author ssah
34   *
35   */
36  public class VisibilityProperty extends NameValueInfoProperty {
37  
38  	boolean visible = true;
39  	
40  	/**
41  	 * @param visible visible
42  	 */
43  	public VisibilityProperty(boolean visible) {
44  		super("", "");
45  	}
46  
47  	/**
48  	 * @param visible the visible to set
49  	 */
50  	public void setVisible(boolean visible) {
51  		this.visible = visible;
52  	}
53  
54  	/* (non-Javadoc)
55  	 * @see com.cosylab.vdct.vdb.NameValueInfoProperty#getVisibility()
56  	 */
57  	public int getVisibility() {
58  		return visible ? NON_DEFAULT_VISIBLE : NEVER_VISIBLE;
59  	}
60  }
