269 private void generateVDBDataInternal(Object dsId, DBDData dbd, DBData db) throws DBException {
270 	
271 	if (dbd!=null && db!=null) {
272 
273 		// extract templates
274 		extractTemplates(dsId, dbd, db);
275 		
276 		// add records, template instances and entries
277 		Enumeration e = db.getStructure().elements();
278 		while (e.hasMoreElements())
279 		{
280 			Object obj = e.nextElement();
281 			if (obj instanceof DBRecordData)
282 			{
283 				DBRecordData dbRecord = (DBRecordData)obj;
284 				addRecord(generateVDBRecordData(dsId, dbd, dbRecord));
285 			}
286 			else if (obj instanceof DBTemplateInstance)
287 			{
288 				DBTemplateInstance dbTemplateInstance = (DBTemplateInstance)obj;
289 				VDBTemplateInstance vti = generateVDBTemplateInstance(dsId, dbTemplateInstance);
290 				if (vti!=null)
291 					addTemplateInstance(vti);
292 			}
293 			else if (obj instanceof DBEntry)
294 			{
295 				addEntry((DBEntry)obj);	
296 			}
297 			
298 		}
299 
300 	}
301 }

348 public void extractTemplates(Object dsId, DBDData dbd, DBData db)
349 {
350 	Enumeration e;
351 	DBTemplate dbTemplate;
352 	e = db.getTemplatesV().elements();
353 	while (e.hasMoreElements()) {
354 		dbTemplate = (DBTemplate)(e.nextElement());
355 
356 		Hashtable templates = VDBData.getInstance(dsId).getTemplates();
357 		if (templates.containsKey(dbTemplate.getId())) {
358 			VDBTemplate t = (VDBTemplate)templates.get(dbTemplate.getId());
359 			Console.getInstance().println("Template with id '"+dbTemplate.getId()+"' ('"+t.getFileName()+"') already exists in repository. Skipping template from file '"+dbTemplate.getFileName()+"'...");
360 			continue;
361 		}
362 		
363 		VDBData vdbData = new VDBData();
364 		vdbData.generateTemplate(dsId, dbd, dbTemplate);
365 	}

368 private void generateTemplate(Object dsId, DBDData dbd, DBTemplate dbTemplate)
369 {
370 	Group root = Group.getRoot(dsId);
371 
372 	// generate vt.group / VDB data
373 	try
374 	{
375 		VDBTemplate vt = new VDBTemplate(dbTemplate.getId(), dbTemplate.getFileName());
376 	
377 		vt.setGroup(new Group(null));
378 		vt.getGroup().setAbsoluteName("");
379 		vt.getGroup().setLookupTable(new Hashtable());
380 	
381 		Group.setRoot(dsId, vt.getGroup());
382 
383 		vt.setComment(dbTemplate.getComment());	
384 		vt.setDescription(dbTemplate.getDescription());
385 		vt.setModificationTime(dbTemplate.getModificationTime());
386 		vt.setVersion(dbTemplate.getVersion());
387 		vt.setIoc(dbTemplate.getIoc());
388 		
389 		generateVDBDataInternal(dsId, dbd, dbTemplate.getData());
390 		DrawingSurface.applyVisualData(dsId, false, vt.getGroup(), dbTemplate.getData(), this);
391 		vt.getGroup().unconditionalValidateSubObjects(false);
392 
393 		Hashtable ports = new Hashtable();
394 		Vector portsV = new Vector();
395 		vt.setPorts(ports);
396 		vt.setPortsV(portsV);
397 
398 		Hashtable macros = new Hashtable();
399 		Vector macrosV = new Vector();
400 		vt.setMacros(macros);
401 		vt.setMacrosV(macrosV);
402 
403 		addPortsAndMacros(dsId, dbTemplate, vt, this);
404 
405 		VDBData.getInstance(dsId).addTemplate(vt);
406 	}
407 	catch (Exception ex)
408 	{
409 		Console.getInstance().println();
410 		Console.getInstance().println("Exception caught while generating '"+dbTemplate.getId()+"' template.");
411 		Console.getInstance().println(ex);
412 		Console.getInstance().println();
413 	}
414 	finally
415 	{
416 		// validate all links
417 		Group.getRoot(dsId).manageLinks(true);
418 		Group.getRoot(dsId).updateFields();
419 		Group.setRoot(dsId, root); 
420 	}
421 }

429 public static void addPortsAndMacros(Object dsId, DBTemplate dbTemplate, VDBTemplate vt, VDBData vdbData, HashMap importedList) {
430 	
431 	// noop (importing into DB w/o editing template data)
432 	if (importedList != null && Group.getEditingTemplateData(dsId) == null)
433 		return;
434 	
435 	Hashtable ports = vt.getPorts();
436 	Vector portsV = vt.getPortsV();
437 	
438 	Hashtable macros = vt.getMacros();
439 	Vector macrosV = vt.getMacrosV();
440 	
441 	Enumeration keys = dbTemplate.getPorts().keys();
442 	while (keys.hasMoreElements())
443 	{
444 		Object key = keys.nextElement();
445 		DBPort port = (DBPort)dbTemplate.getPorts().get(key);
446 		VDBPort vdbPort = new VDBPort(vt, port);	
447 		
448 		// skip
449 		if (Group.getRoot(dsId).getSubObject(vdbPort.getName())!= null)
450 		{
451 			Console.getInstance().println("WARNING: port with name '" + vdbPort.getName() + "' already exists, skipping its redefinition...");
452 			continue;
453 		}
454 		
455 		// has visual
456 		if (port.isHasVisual())
457 		{
458 			Port visualPort = new Port(vdbPort, Group.getRoot(dsId),
459 								  port.getX(), port.getY());
460 			visualPort.setColor(port.getColor());
461 			visualPort.setMode(port.getMode());
462 			visualPort.setTextPositionNorth(port.isNamePositionNorth());
463 
464 			// delegate defaultVisibility
465 			vdbPort.setVisibility(port.getDefaultVisibility());
466 			
467 			Group.getRoot(dsId).addSubObject(vdbPort.getName(), visualPort);
468 			if (importedList != null) importedList.put(vdbPort.getName(), visualPort);
469 		}
470 
471 		ports.put(key, vdbPort);
472 		portsV.addElement(vdbPort);
473 	}
474 
475 
476 	keys = dbTemplate.getMacros().keys();
477 	while (keys.hasMoreElements())
478 	{
479 		Object key = keys.nextElement();
480 		DBMacro macro = (DBMacro)dbTemplate.getMacros().get(key);
481 		VDBMacro vdbMacro = new VDBMacro(vt, macro);	
482 
483 		// skip
484 		if (Group.getRoot(dsId).getSubObject(vdbMacro.getName())!= null)
485 		{
486 			Console.getInstance().println("WARNING: macro with name '" + vdbMacro.getName() + "' already exists, skipping its redefinition...");
487 			continue;
488 		}
489 
490 		// has visual
491 		if (macro.isHasVisual())
492 		{
493 			Macro visualMacro = new Macro(vdbMacro, Group.getRoot(dsId),
494 								  macro.getX(), macro.getY());
495 			visualMacro.setColor(macro.getColor());
496 			visualMacro.setMode(macro.getMode());
497 			visualMacro.setTextPositionNorth(macro.isNamePositionNorth());
498 			
499 			// delegate defaultVisibility
500 			vdbMacro.setVisibility(macro.getDefaultVisibility());
501 			
502 			Group.getRoot(dsId).addSubObject(vdbMacro.getName(), visualMacro);
503 			if (importedList != null) importedList.put(vdbMacro.getName(), visualMacro);
504 		}
505 
506 		macros.put(key, vdbMacro);
507 		macrosV.addElement(vdbMacro);
508 		
509 	}
510 
511 	DrawingSurface.applyPortAndMacroConnectors(dsId, dbTemplate.getData(), vdbData);
512 }

609 	vdbRecord.setComment(dbRecord.getComment());
610 
611 	// order preservation
