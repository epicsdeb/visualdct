45  /**
46   * A default LinkTypeConfig plugin. 
47   * It reads link type configuration from an XML file. The file name is defined by VDCT_LINK_CONFIG_FILE filename located
48   * in user home directory (or in VDCT_CONFIG_DIR dir). Use VDCT_LINK_CONFIG_FILE_ENV environment variable to override this default setting.
49   * An example of XML file:
50   * <pre>
51   * {@code
52   * <?xml version="1.0" encoding="UTF-8"?>
53   * 
54   * <links>
55   *    <link type="CONSTANT" pattern=".*" default="" description="CONSTANT" />
56   *    <link type="PV_LINK" pattern=".*" default="" description="PV_LINK" />
57   *    <link type="VME_IO" pattern="#C\d+ S\d+ @.*" default="#C0 S0( )*( @.*)?" description="VME_IO - #Ccard Ssignal @parm"  />
58   *    <link type="CAMAC_IO" pattern="#B\d+ C\d+ N\d+ A\d+ F\d+( )*( @.*)?" default="CAMAC_IO - #B0 C0 N0 A0 F0 @" description="#Bbranch Ccrate Nstation Asubaddress Ffunction @parm" />
59   *    <link type="AB_IO" pattern="#L\d+ A\d+ C\d+ S\d+( )*( @.*)?" default="#L0 A0 C0 S0 @" description="AB_IO - #Llink Aadapter Ccard Ssignal @parm" />
60   *    <link type="GPIB_IO" pattern="#L\d+ A\d+( )*( @.*)?" default="#L0 A0 @" description="GPIB_IO - #Llink Aaddr @parm"  />
61   *    <link type="BITBUS_IO" pattern="#L\d+ N\d+ P\d+ S\d+( )*( @.*)?" default="BITBUS_IO - @L0 N0 P0 S0 @" description="#Llink Nnode Pport Ssignal @parm"  />
62   *    <link type="INST_IO" pattern="@.*" default="@" description="INST_IO - @"  />
63   *    <link type="BBGPIB_IO" pattern="#L\d+ B\d+ G\d+( )*( @.*)?" default="#L0 B0 G0 @" description="BBGPIB_IO - #Llink Bbbaddr Ggpibaddr @parm"  />
64   *    <link type="RF_IO" pattern="#R\d+ M\d+ D\d+ E\d+( )*( @.*)?" default="#R0 M0 D0 E0 @" description="RF_IO - #Rcryo Mmicro Ddataset Eelement"  />
65   *    <link type="VXI_IO" pattern="#V\d+ (C\d+)?+  S\d+( )*( @.*)?" default="#V0 C0 S0 @" description="VXI_IO - #Vframe Cslot Ssignal @parm"  />
66   * </links>
67   * }
68   * </pre>
69   * @author Matej Sekoranja
70   */
71  public class LinkTypeConfig implements LinkTypeConfigPlugin

73  	
74  	public static final String VDCT_LINK_CONFIG_FILE = ".vdctlinks.xml";
75  	public static final String VDCT_LINK_CONFIG_FILE_ENV = "VDCT_LINK_CONFIG";
76  	
77  	/**
78  	 * @see com.cosylab.vdct.plugin.config.LinkTypeConfigPlugin#getLinkTypeConfig()
79  	 */
80  	public Hashtable getLinkTypeConfig()
81  	{
82  
83  		String fileName = null;
84  		Document doc = null;
85  		try
86  		{
87  		    fileName = Constants.getConfigFile(VDCT_LINK_CONFIG_FILE, VDCT_LINK_CONFIG_FILE_ENV);
88  		
89  			// is file does not exists, load default file
90  			if (!(new java.io.File(fileName).exists()))
91  			{
92  				com.cosylab.vdct.Console.getInstance().println("[LinkTypeConfigPlugin] No link type configuration file found.");
93  				//fileName = getClass().getResource("/"+Constants.CONFIG_DIR+VDCT_LINK_CONFIG_FILE).getFile();
94  				return null;
95  			}
96  
97  			doc = XMLManager.readFileDocument(fileName);
98  		}
99  		catch (Exception e)
100 		{
101 			com.cosylab.vdct.Console.getInstance().println("[LinkTypeConfigPlugin] Failed to open link type configuration file '"+fileName+"'.");
102 			com.cosylab.vdct.Console.getInstance().println(e);
103 			return null;
104 		}
105 	
106 		if (doc==null)
107 		{
108 			com.cosylab.vdct.Console.getInstance().println("[LinkTypeConfigPlugin] Invalid link type configuration file '"+fileName+"'.");
109 			return null;
110 		}
111 
112 		Hashtable table = new Hashtable();
113 						
114 		int count = 0;
115 		//Node node = doc.getElementsByTagName("links").item(0);
116 		NodeList nodes = doc.getElementsByTagName("link");
117 		for (int i=0; i<nodes.getLength(); i++)
118 		{
119 			Element e = (Element)nodes.item(i); 
120 			String type = e.getAttribute("type");
121 			String pattern = e.getAttribute("pattern");
122 			String def = e.getAttribute("default");
123 			String desc = e.getAttribute("description");
124 			if (type!=null && pattern!=null && def!=null && desc!=null)
125 			{
126 				try
127 				{
128 					table.put(type, new Object[] { Pattern.compile(pattern), def, desc });
129 					count++;
130 				}
131 				catch (PatternSyntaxException pse)
132 				{
133 					com.cosylab.vdct.Console.getInstance().println("[LinkTypeConfigPlugin] Invalid pattern '"+pattern+"' for link type '"+type+"'. Skipping...");
134 					com.cosylab.vdct.Console.getInstance().println(pse);
135 					com.cosylab.vdct.Console.getInstance().println();
136 				} 
137 			}
138 			else if (type!=null)
139 				com.cosylab.vdct.Console.getInstance().println("[LinkTypeConfigPlugin] Invalid configuration for link type '"+type+"'. Skipping...");
140 			else
141 				com.cosylab.vdct.Console.getInstance().println("[LinkTypeConfigPlugin] Invalid configuration found. Skipping....");
142 		}
143 		com.cosylab.vdct.Console.getInstance().println("[LinkTypeConfigPlugin] Loaded "+count+" link type configuration(s).");
144 		
145 		return table;
146 	}
