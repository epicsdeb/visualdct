45  /**
46   * This NameConfigPlugin example checks the record names using a pattern loaded from an xml file. 
47   * The file name is defined by VDCT_NAME_CONFIG_FILE filename located in user home directory
48   * (or in VDCT_CONFIG_DIR dir). Use VDCT_NAME_CONFIG_FILE_ENV environment variable to override
49   * this default setting.
50  
51   * An example of XML file:
52   * <pre>
53   * {@code
54   * <?xml version="1.0" encoding="UTF-8"?>
55   * 
56   * <name pattern="VDCT_.*" default="VDCT_Name1" description="Must start with VDCT_"/>
57   * 
58   * Default and description parameters are optional.
59   * }
60   * </pre>
61   * 
62   * @author ssah
63   */
64  public class NameConfigExample implements NameConfigPlugin {
65  
66  	public static final String VDCT_NAME_CONFIG_FILE = ".vdctname.xml";
67  	public static final String VDCT_NAME_CONFIG_FILE_ENV = "VDCT_LINK_CONFIG";
68  	
69  	private boolean running = false;
70  	
71  	private Pattern pattern = null;
72  	private String defaultName = null;
73  	private String description = null;
74  	
75  	/* (non-Javadoc)
76  	 * @see com.cosylab.vdct.plugin.config.NameConfigPlugin#checkValidity(java.lang.String)
77  	 */
78  	public String checkValidity(String name) {
79  		if (pattern != null && !pattern.matcher(name).matches()) {
80  			return "The name does not match the pattern: " + pattern.pattern();
81  		}
82  		return null;
83  	}
84  
85  	/**
86  	 * Returns the default name.    
87  	 */
88  	public String getDefaultName() {
89  		return defaultName;
90  	}
91  	
92  	/* (non-Javadoc)
93  	 * @see com.cosylab.vdct.plugin.config.NameConfigPlugin#getNameDescription()
94  	 */
95  	public String getNameDescription() {
96  		return description;
97  	}
98  
99  	/* (non-Javadoc)
100 	 * @see com.cosylab.vdct.plugin.config.NameConfigPlugin#isRunning()
101 	 */
102 	public boolean isRunning() {
103 		return running;
104 	}
105 
106 	/* (non-Javadoc)

108 	 */
109 	public void destroy() {
110 
111 	}

118 	}

123 	public String getDescription() {
124 		return "NameConfigPlugin which loads name validation pattern from an XML file.";
125 	}

131 		return "Name Configuration";
132 	}
133

136 	 */
137 	public String getVersion() {
138 		return "0.1";

141 	/* (non-Javadoc)
142 	 * @see com.cosylab.vdct.plugin.Plugin#init(java.util.Properties, com.cosylab.vdct.plugin.PluginContext)
143 	 */
144 	public void init(Properties properties, PluginContext context) {
145 
146 		String fileName = null;
147 		Document doc = null;
148 		try
149 		{
150 			fileName = Constants.getConfigFile(VDCT_NAME_CONFIG_FILE, VDCT_NAME_CONFIG_FILE_ENV);
151 
152 			// is file does not exists, load default file
153 			if (!(new java.io.File(fileName).exists()))
154 			{
155 				Console.getInstance().println("[NameConfigPlugin] No name configuration file found.");
156 				return;
157 			}
158 
159 			doc = XMLManager.readFileDocument(fileName);
160 		} catch (Exception e) {
161 			Console.getInstance().println("[NameConfigPlugin] Failed to open name configuration file '"+fileName+"'.");
162 			Console.getInstance().println(e);
163 			return;
164 		}
165 
166 		if (doc==null) {
167 			Console.getInstance().println("[NameConfigPlugin] Invalid name configuration file '"+fileName+"'.");
168 			return;
169 		}
170 
171 		NodeList nodes = doc.getElementsByTagName("name");
172 		for (int i=0; i<nodes.getLength(); i++)
173 		{
174 			Element e = (Element)nodes.item(i); 
175 			String patternString = e.getAttribute("pattern");
176 			defaultName = e.getAttribute("default");
177 			description = e.getAttribute("description");
178 			if (patternString != null)
179 			{
180 				try {
181 					pattern = Pattern.compile(patternString);
182 				} catch (PatternSyntaxException pse) {
183 					Console.getInstance().println("[NameConfigPlugin] Invalid pattern '"+pattern+"'. Skipping...");
184 					Console.getInstance().println(pse);
185 					Console.getInstance().println();
186 				}
187 			} else {
188 				Console.getInstance().println("[NameConfigPlugin] Invalid name configuration file '"+fileName+"'.");
189 			}
190 		}
191 	}
192

197 		running = true;
198 	}
199

202 	 */
203 	public void stop() {
204 		running = false;
205 	}
