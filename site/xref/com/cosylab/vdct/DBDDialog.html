121 	 * Returns an attribute value for the cell at <I>row</I>

123 	 *

125 	 * @param   column          the column whose value is to be looked up

129 	public Object getValueAt(int row, int column) {
130 		DBDEntry entry = (DBDEntry)DataProvider.getInstance().getCurrentDBDs().get(row);
131 		switch (column) {
132 			case 0: return entry.getValue();
133 			case 1: return entry.getFile().getAbsoluteFile();
134 			case 2: return new Boolean(entry.isRelative());
135 			case 3: return new Boolean(entry.getSavesToFile());
136 			default:
137 				return null;
138 		}
139 	}

152 	/**
153 	 * Sets the object value for the cell at <I>column</I> and
154 	 * <I>row</I>.  <I>aValue</I> is the new value.  This method
155 	 * will generate a tableChanged() notification.
156 	 *
157 	 * @param   aValue          the new value.  This can be null.
158 	 * @param   row             the row whose value is to be looked up
159 	 * @param   column          the column whose value is to be looked up
160 	 */
161 
162 	public void setValueAt(Object aValue, int row, int column) {
163 		Vector dbds = DataProvider.getInstance().getCurrentDBDs();
164 		DBDEntry entry = (DBDEntry)dbds.get(row);
165 	
166 		switch (column) {
167 			case 2:
168 				entry.setRelative(((Boolean)aValue).booleanValue());
169 				fireTableCellUpdated(row, 0);
170 				break;
171 				
172 			case 3:
173 				entry.setSavesToFile(((Boolean)aValue).booleanValue());
174 				break;
175 				
176 			case 0:
177 				if (aValue==null)
178 				{
179 					dbds.removeElementAt(row);
180 					//update entire table
181 					getScrollPaneTable().tableChanged(new TableModelEvent(getScrollPaneTable().getModel()));
182 					return;
183 				}	
184 				
185 				entry.setValue(aValue.toString());
186 			
187 				File f = entry.getFile();
188 				if (f.exists())
189 				{
190 					GetGUIInterface cmd = (GetGUIInterface)CommandManager.getInstance().getCommand("GetGUIMenuInterface");
191 					try {
192 						cmd.getGUIMenuInterface().importDBD(f);
193 					} catch (java.io.IOException e) {
194 						Console.getInstance().println();
195 						Console.getInstance().println("o) Failed to load DBD file: '"+f.getAbsolutePath()+"'.");
196 						Console.getInstance().println(e);
197 					}
198 				}
199 		
200 		}
201 		
202 		// generate notification
203 		fireTableCellUpdated(row, column);
204 	}

283 	chooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
284 	chooser.setDialogType(JFileChooser.OPEN_DIALOG);
285 	int retval = chooser.showOpenDialog(this);
286 
287 	if(retval == JFileChooser.APPROVE_OPTION) {
288 	    java.io.File theFile = chooser.getSelectedFile();
289 	    if(theFile != null) {
290 			theFile = theFile.getAbsoluteFile();
291 	    				
292 		    GetGUIInterface cmd = (GetGUIInterface)CommandManager.getInstance().getCommand("GetGUIMenuInterface");
293 		    try {
294 		    	
295 		    	DBDEntry entry = new DBDEntry(theFile.getPath());
296 				entry.setRelative(!chooser.getJCheckBoxAbsoluteDBD().isSelected());
297 		    	
298 		    	Vector dbds = DataProvider.getInstance().getCurrentDBDs();
299 				//	replace
300 				if (dbds.contains(entry)) dbds.remove(entry);
301 				dbds.addElement(entry);
302 				
303 				getScrollPaneTable().tableChanged(new TableModelEvent(getScrollPaneTable().getModel()));
304 				
305 	  		 	cmd.getGUIMenuInterface().importDBD(entry.getFile());
306 		    } catch (java.io.IOException e) {
307 			    Console.getInstance().println();
308 			    Console.getInstance().println("o) Failed to import DBD file: '"+theFile.getAbsolutePath()+"'.");
