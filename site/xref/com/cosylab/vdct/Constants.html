58   	/**
59   	 * Generate full config file path using default algoritm: override JVM property, user home, sys home.
60   	 * @param fileName	config file name (w/o path)
61   	 * @param overrideProperty	override property (JVM parameter) which overrides default config file name (full path), can be <code>null</code>. 
62   	 * @return full config file path (file might not exist).
63   	 */
64   	public static String getConfigFile(final String fileName, final String overrideProperty) {
65   	    
66   	    // first check override property, if available
67   	    if (overrideProperty != null) {
68  			String overriden = System.getProperty(overrideProperty);
69  			if (overriden != null)
70  			    return overriden;
71   	    }
72  		
73  		// check user home
74  		String file = System.getProperty("user.home") + "/" + fileName;
75  		if (new java.io.File(file).exists())
76  		    return file;
77  		
78  		// return sys dir
79  		return VDCT_CONFIG_DIR + "/" + fileName;
80   	}
