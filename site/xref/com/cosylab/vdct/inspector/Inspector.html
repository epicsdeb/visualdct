685  * @param object com.cosylab.vdct.inspector.Inspectable
686  * @param raise
687  */
688 public void inspectObject(Inspectable object, boolean raise) {
689 
690 	closeCellEditors(true);
691 	if (inspectedObject==object) return;
692 	inspectedObject=object;
693 
694 	if (inspectedObject==null) {
695 		getHelpLabel().setText("No object selected");
696 		getCommentTextArea().setProperty(null);
697 		tableModel.setDataObject(null);
698 		setTitle("Inspector");
699 		if (raise) setVisible(true);
700 		return;
701 	}
702 
703 	boolean isOutsider = !objs.contains(object);
704 	if (isOutsider)
705 		getObjectComboBox().addItem(object);
706 		
707 	// !!! object must be in list !!!
708 	getObjectComboBox().setSelectedItem(object);
709 
710 	if (isOutsider)
711 		toRemove = object;
712 	
713 	//com.cosylab.vdct.Console.getInstance().println("Inspecting: "+object);
714 
715 	setTitle("Inspector - "+inspectedObject.getName());
716 	tableModel.setDataObject(inspectedObject);
717 
718 	getCommentTextArea().setProperty(inspectedObject.getCommentProperty());
719 	getCommentTextArea().setCaretPosition(0);
720 
721 	// setup tabs
722 	initializeTabs(object.getModeNames());
723 	
724 	if (raise) setVisible(true);
725 	
726 	if (!updatingList)
727 	{
728 		if (object instanceof VisibleObject) {
729 			VisibleObject visible = (VisibleObject)object;
730 		    ViewState.getInstance(visible.getDsId()).setAsHilited(visible);
731 		    DsManager.getDrawingSurface(dsId).repaint(true);
732 		}
733 	}
734 }
