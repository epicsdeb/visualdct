221 	public void sortRowsByColumn(int column) {
222 
223 		int visibleIndex = modelToPropertiesColumnIndex[column];
224 		int newSortedColumn = visibleToBaseColumn(visibleIndex);
225 
226 		int newSortedSplitIndex = column - propertiesToModelColumnIndex[visibleIndex];
227 
228 		boolean sameColumn = (sortedColumn == newSortedColumn && sortedSplitIndex == newSortedSplitIndex); 
229 		sortedOrderAsc = sameColumn ? !sortedOrderAsc : true;
230 		sortedColumn = newSortedColumn;
231 		sortedSplitIndex = newSortedSplitIndex;
232 
233 		int[] order = PropertyComparator.getOrder(model, true, column, sortedOrderAsc);
234         setRowOrder(order);
235 		refreshModel();
236 	}

298 		// Check if the cell is part of the comment, if so return it.
299 		if (row >= 0 && column >= 0) {
300 			InspectableProperty property = model[row][column];
301 			InspectableProperty baseProperty = property;
302 			if (baseProperty instanceof SplitPropertyPart) {
303 			    baseProperty = ((SplitPropertyPart)baseProperty).getOwner().getOwner();
304 			}
305 			
306 			if (baseProperty instanceof CommentProperty) {
307 				String value = property.getValue();
308 				return (value == null) ? "" : value;
309 			}
310 		}
311 		return null;
312 	}

467 	protected boolean internalSetValueAt(Object value, int row, int column) {
468 		
469 		boolean change = false;
470 		
471 		InspectableProperty property = model[row][column];
472 
473 		int propertiesColumn = modelToPropertiesColumnIndex[column];
474 		
475 		if (property instanceof SplitPropertyPart) {
476 			/* If this split property is part of creator property, set the new created property as the owner
477 			 * of this group.  
478 			 */
479 			SplitPropertyGroup propertyGroup = ((SplitPropertyPart)property).getOwner();
480 			InspectableProperty baseProperty = propertyGroup.getOwner(); 
481 			
482 			if (baseProperty instanceof CreatorProperty) {
483 				// Just call to create a new property, value doesn't matter as it will be overwritten. 
484 				change |= super.internalSetValueAt(value, row, propertiesColumn);
485                 InspectableProperty newProperty = ((CreatorProperty)baseProperty).getCreatedProperty();
486     			// If a new property has been created, set it.
487 				if (newProperty != null) { 
488 					propertyGroup.setOwner(newProperty);
489 	                property.setValue(value.toString());
490 	                change = true;
491 				}
492 			} else {
493                 property.setValue(value.toString());
494                 change = true;
495 			}
496 		} else {
497 			/* If this isn't a split part, just call the base function and update the model data structure if a new
498 			 * property has been created by creator property.
499 			 */
500 			change |= super.internalSetValueAt(value, row, propertiesColumn);
501 			if (property instanceof CreatorProperty) {
502 				
503                 InspectableProperty newProperty = ((CreatorProperty)property).getCreatedProperty();
504     			// If a new property has been created, replace it.
505 				if (newProperty != null) { 
506 					model[row][column] = newProperty;
507 				}
508 			}
509 		}
510 		
511 		/* TASK:NONMODALSPR
512 		 * On name change, if the whole table is not refreshed, at least splitting has to be recalculated as the
513 		 * field values that point to the name change.
514 		 */
515 		/*
516 		if (propertiesColumn == getNamesColumn()) {
517    			refreshModel();
518 		}
519 		*/
520 		
521 		return change;
522 	}

546     /**
547      *
548      * @param rows rows
549      * @param columns columns
550      */
551     public void extendCounters(int[] rows, int[] columns) {
552 		
553 		UndoManager.getInstance(dsId).startMacroAction();
554 		for (int c = 0; c < columns.length; c++) {
555 			String firstEntry = model[rows[0]][columns[c]].getValue();
556 			String secondEntry = model[rows[1]][columns[c]].getValue();
557 			String baseString = DBSheetSplitCol.removeValueAtEnd(firstEntry);
558 			int firstValue = DBSheetSplitCol.extractValueAtEnd(firstEntry);
559 			int secondValue = DBSheetSplitCol.extractValueAtEnd(secondEntry);
560 
561 			// Assume base at 0 and step 1 if values are absent.
562 			if (firstValue == -1) {
563 				firstValue = 0;
564 			}
565 			if (secondValue == -1 || secondValue == firstValue) {
566 				secondValue = firstValue + 1;
567 			}
568 			
569 			int value = firstValue;
570 			int step = secondValue - firstValue;
571 
572 			for (int r = 0; r < rows.length; r++) {
573 				internalSetValueAt(baseString + value, rows[r], columns[c]);
574 				value += step;
575 			}
576 		}
577 		// Changes can affect whole rows as certain field validity states depend on other fields of the inspectable.
578 		for (int r = 0; r < rows.length; r++) {
579 			fireTableRowsUpdated(rows[r], rows[r]);
580 		}
581         UndoManager.getInstance(dsId).stopMacroAction();
582 	}

626 	private void refreshModel() {
627 
628 		int viewColumnCount = super.getColumnCount();
629 		int viewRowCount = super.getRowCount();
630 		updateSplitData();
631         
632 		model = new InspectableProperty[viewRowCount][];
633 		for (int i = 0; i < viewRowCount; i++) {
634 			model[i] = new InspectableProperty[modelColumnCount];
635 		}
636 		
637 		// Set the model column names and model data according to split columns.
638 		modelColumnNames = new String[modelColumnCount];
639 		for (int j = 0; j < viewColumnCount; j++) {
640 
641 		    String name = super.getColumnId(j);
642 			DBSheetSplitCol split = propertiesColumnSplitData[super.visibleToBaseColumn(j)]; 
643 			int modelJ = propertiesToModelColumnIndex[j];
644 
645 			if (split != null) {
646 
647 	    		String colName = null;
648 		    	int parts = split.getParts();
649 				for (int p = 0; p < parts; p++) {
650 					colName = "";
651 					if (p == 0) {
652 						colName += "<";
653 					}
654 					colName += name + "[" + (p + 1) + "]";
655 					if (p == parts - 1) {
656 						colName += ">";
657 					}
658 			    	modelColumnNames[modelJ + p] = colName;
659 				}
660 				
661 				SplitPropertyGroup[] splitGroups = split.getSplitGroups();
662 				for (int i = 0; i < viewRowCount; i++) {
663 					//SplitPropertyGroup group = new SplitPropertyGroup(super.getPropertyAt(i, j), split);
664 					SplitPropertyGroup group = splitGroups[visibleToBaseRow(i)];
665 					
666 					for (int p = 0; p < parts; p++) {
667 						model[i][modelJ + p] = group.getPart(p);
668 					}
669 				}
670 			} else {
671 		    	modelColumnNames[modelJ] = name;
672 				for (int i = 0; i < viewRowCount; i++) {
673 					model[i][modelJ] = super.getPropertyAt(i, j);
674 				}
675             }
676 		}
677 	}

679 	private void updateSplitData() {
680 
681 		int propertiesRowCount = getPropertiesRowCount();
682 		int propertiesColumnCount = getPropertiesColumnCount();
683 		
684 		for (int j = 0; j < propertiesColumnCount; j++) {
685 			DBSheetSplitCol split = propertiesColumnSplitData[j];
686 			if (split != null) {
687 				
688                 // Check if current data is out of date.
689 				boolean refresh = false;
690 				SplitPropertyGroup[] splitGroups = split.getSplitGroups();
691 				if (splitGroups != null && splitGroups.length == propertiesRowCount) {
692 					for (int i = 0; i < propertiesRowCount; i++) {
693 						if (splitGroups[i] == null || !StringUtils.emptyEquals(splitGroups[i].getValue(), getPropertyValue(i, j))) {
694 							refresh = true;
695 							break;
696 						}
697 					}
698 				} else {
699 					refresh = true;
700 				}
701 				
702 				if (refresh) {
703 					splitGroups = new SplitPropertyGroup[propertiesRowCount];
704 					split.setSplitGroups(splitGroups);
705 
706 					/* Calculate the number of required columns to which the current must be split into.
707 					 */
708 					int maxParts = 1;
709 					int parts = 0;
710 					for (int i = 0; i < propertiesRowCount; i++) {
711 						parts = SplitPropertyGroup.getPartsCount(getPropertyValue(i, j), split);
712 						maxParts = Math.max(maxParts, parts);
713 					}
714 					split.setParts(maxParts);
715 					
716 					for (int i = 0; i < propertiesRowCount; i++) {
717 					    splitGroups[i] = new SplitPropertyGroup(getProperty(i, j), split);
718 					}
719 				}
720 			}
721 		}
722 
723 		int viewColumnCount = super.getColumnCount();
724 		
725 		// Create split translation tables.
726 		DBSheetSplitCol splitData = null;
727 		modelColumnCount = 0;
728 		for (int j = 0; j < viewColumnCount; j++) {
729 			splitData = propertiesColumnSplitData[super.visibleToBaseColumn(j)];
730 			modelColumnCount += (splitData != null) ? splitData.getParts() : 1;
731 		}
732 		
733 		propertiesToModelColumnIndex = new int[viewColumnCount];
734 		modelToPropertiesColumnIndex = new int[modelColumnCount];
735 		
736 		int col = 0;
737 		for (int j = 0; j < viewColumnCount; j++) {
738 		    propertiesToModelColumnIndex[j] = col;
739 			splitData = propertiesColumnSplitData[super.visibleToBaseColumn(j)];
740 		    int parts = (splitData != null) ? splitData.getParts() : 1;
741 		    for (int p = 0; p < parts; p++) {
742 		    	modelToPropertiesColumnIndex[col] = j;
743 		    	col++;
744 		    }
745 		}
746 	}

748 	private void recallSplitData() {
749 		DBSheetView record = getViewRecord();
750 
751 		// Set the split data and refresh the model.
752     	for (int j = 0; j < getPropertiesColumnCount(); j++) {
753     		propertiesColumnSplitData[j] = null;
754     	}
755     	DBSheetSplitCol[] splitColumns = record.getSplitColumns();
756     	if (splitColumns != null) {
757     		for (int j = 0; j < splitColumns.length; j++) {
758     			int index = getPropertiesColumnIndex(splitColumns[j].getName());
759     			if (index >= 0) {
760     				propertiesColumnSplitData[index] = splitColumns[j];
761     			}
762     		}
763     	}
764 		refreshModel();
765     	
766     	DBSheetSplitCol[] recentSplits = record.getRecentSplits();
767     	if (recentSplits != null) {
768     		recentSplitData.clear();
769     		for (int j = 0; j < recentSplits.length; j++) {
770         		recentSplitData.add(recentSplits[j]);
771     		}
772     	}
773 	}

775 	private void recallViewData() {
776 		DBSheetView record = getViewRecord();
777 
778 		DBSheetRowOrder rowOrder = record.getRowOrder();
779         if (rowOrder != null) {
780         	String orderedColumnName = rowOrder.getColumnName();
781         	int index = getPropertiesColumnIndex(orderedColumnName);
782         	if (index >= 0) {
783         		int visibleIndex = baseToVisibleColumn(sortedColumn);
784         		if (visibleIndex >= 0) {
785         			sortedSplitIndex = rowOrder.getColumnSplitIndex();
786         			int modelIndex = propertiesToModelColumnIndex[visibleIndex] + sortedSplitIndex;
787         			int[] order = PropertyComparator.getOrder(model, true, modelIndex, sortedOrderAsc);
788         			setRowOrder(order);
789         			refreshModel();
790         		}
791         	}
792         }

798     protected void storeSplitData() {
799 		DBSheetView record = getViewRecord();
800 		
801 		boolean defaultNoColumnSplit = true;
802 		int propertiesColumnCount = getPropertiesColumnCount();
803         for (int j = 0; j < propertiesColumnCount; j++) {
804         	if (propertiesColumnSplitData[j] != null) {
805         		defaultNoColumnSplit = false;
806         		break;
807         	}
808         }
809         
810         if (!defaultNoColumnSplit) {
811             DBSheetSplitCol[] splitColumns = null;
812         	Vector splitColumnVector = new Vector();
813         	for (int j = 0; j < propertiesColumnCount; j++) {
814         		if (propertiesColumnSplitData[j] != null) {
815             		splitColumnVector.add(propertiesColumnSplitData[j]);
816         		}
817         	}
818         	splitColumns = new DBSheetSplitCol[splitColumnVector.size()]; 
819             splitColumnVector.copyInto(splitColumns);
820     		record.setSplitColumns(splitColumns);
821         } else {
822     		record.setSplitColumns(null);
823         }
824 
825 		boolean defaultNoRecentSplits = recentSplitData.isEmpty();
826         if (!defaultNoRecentSplits) {
827         	DBSheetSplitCol[] recentSplits = new DBSheetSplitCol[recentSplitData.size()]; 
828         	recentSplitData.copyInto(recentSplits);
829     		record.setRecentSplits(recentSplits);
830         } else {
831     		record.setRecentSplits(null);
832         }
833 	}
