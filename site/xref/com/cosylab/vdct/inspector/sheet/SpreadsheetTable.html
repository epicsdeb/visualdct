181 	/* Refit column widths for columns that have a default width.
182 	 * 
183 	 * @see javax.swing.JTable#tableChanged(javax.swing.event.TableModelEvent)
184 	 */
185 	public void tableChanged(TableModelEvent e) {
186 		super.tableChanged(e);
187 
188 		// On INSERT and DELETE the widths are updated trough other calls. 
189 		if (e.getType() == TableModelEvent.UPDATE) {
190 
191 			if (getFont() != null) {
192 				FontMetrics metrics = graphics.getFontMetrics(getFont());
193 				TableColumnModel colModel = getColumnModel();
194 
195 				int startColumn = e.getColumn();
196 				int endColumn = e.getColumn() + 1;
197 				if (e.getColumn() == TableModelEvent.ALL_COLUMNS) {
198 					startColumn = 0;
199 					endColumn = colModel.getColumnCount();
200 				}
201 
202 				int startRow = 0;
203 				int endRow = sprModel.getRowCount();
204 				// Faster but cannot shrink widths back.
205 				// startRow = e.getFirstRow();
206 				// endRow = e.getLastRow() + 1;
207 				if (e.getColumn() == TableModelEvent.ALL_COLUMNS) {
208 					startColumn = 0;
209 					endColumn = colModel.getColumnCount();
210 				} 
211 
212 				SpreadsheetColumn column = null;
213 				for (int c = startColumn; c < endColumn; c++) {
214 					column = (SpreadsheetColumn)colModel.getColumn(c);
215 					if (column.isDefaultWidth()) {
216 
217 						int colWidth = (c == 0) ? firstColumnWidth : minColumnWidth;
218 
219 						column = (SpreadsheetColumn)getColumnModel().getColumn(c);
220 
221 						String value = column.getHeaderValue().toString();
222 						colWidth = Math.max(colWidth, metrics.stringWidth(value) + columnMargin);
223 
224 						for (int r = startRow; r < endRow; r++) {
225 							value = getValueAt(r, c).toString();
226 							colWidth = Math.max(colWidth, metrics.stringWidth(value) + columnMargin);
227 						}
228 						// Faster but cannot shrink widths back.
229 						/*
230 						if (column.getPreferredWidth() < colWidth) {
231 							column.setPreferredWidth(colWidth);
232 						}
233 						 */
234 						column.setPreferredWidth(colWidth);
235 					}
236 				}
237 			}
238 		}
239 	}

293 
294 		int colCount = sprModel.getColumnCount();
295 		TableColumnModel colModel = getColumnModel();
296 		SpreadsheetColumn column = null;
297 
298 		for (int i = 0; i < colCount; i++) {
299 			column = (SpreadsheetColumn)colModel.getColumn(i);

306     /**
307      *
308      * @param columnIndex columnIndex
309      */
310     public void setColumnSizeToFit(int columnIndex) {
311 
312 		FontMetrics metrics = graphics.getFontMetrics(getFont());
313 
314 		int colWidth = (columnIndex == 0) ? firstColumnWidth : minColumnWidth;
315 
316 		SpreadsheetColumn column = (SpreadsheetColumn)getColumnModel().getColumn(columnIndex);
317 
318 		String value = column.getHeaderValue().toString();
319 		colWidth = Math.max(colWidth, metrics.stringWidth(value) + columnMargin);
320 
321 		int rowCount = getRowCount();
322 		for (int j = 0; j < rowCount; j++) {
323 			value = getValueAt(j, columnIndex).toString();
324 			colWidth = Math.max(colWidth, metrics.stringWidth(value) + columnMargin);
325 		}
326 		column.setPreferredWidth(colWidth);
327 		column.setDefaultWidth(true);
328 	}

346 	public void actionPerformed(ActionEvent event) {
347 		String action = event.getActionCommand();
348 		Object source = event.getSource();
349 
350 		if (action.equals(hide)) {
351 			if (popupModelColumn >= 0) {
352 				int propertyColumn = sprModel.getPropertyColumn(popupModelColumn);
353 				propertiesColumnMenuItem[propertyColumn].setSelected(false);
354 				sprModel.setColumnsVisibility(new int[] {popupModelColumn}, false);
355 				setDefaultWidthColumnsToFit();
356 				clearSelection();
357 			}
358 		} else if (action.equals(whitespaces) || action.equals(customPattern)) {
359 
360 			int index = popupModelColumn;
361 			if (index >= 0) {
362 				DBSheetSplitCol split = null; 
363 				if (action.equals(customPattern)) {
364 					CustomSplitDialog dialog = inspector.getCustomSplitDialog();
365 					dialog.setLocationRelativeTo(inspector);
366 
367 					Vector recentSplitData = sprModel.getRecentSplitData();
368 					if (recentSplitData.size() > 0) {
369 						split = (DBSheetSplitCol)recentSplitData.get(0);
370 					} else {
371 						split = DBSheetSplitCol.getWhitespaceSplitData();
372 					}
373 					dialog.setSplitData(split);
374 					/* Add the data from the first row of the splitting column as starting test example. If no visible
375 					 * rows, add from the first among all.
376 					 */
377 					dialog.setTestExample((sprModel.getRowCount() >= 0) ? sprModel.getModelValue(0, index) : "");
378 					dialog.setVisible(true);
379 					split = dialog.getSplitData();
380 					if (split != null) {
381 						recentSplitData.add(0, split);
382 						while (recentSplitData.size() > sprModel.getRecentSplitDataMaxCount()) {
383 							recentSplitData.remove(recentSplitData.size() - 1);
384 						}
385 						refreshRecentSplitsMenu();
386 					}
387 
388 				} else {
389 					split = DBSheetSplitCol.getWhitespaceSplitData();
390 				}
391 				if (split != null) {
392 					sprModel.splitColumn(split, index);
393 					refreshRecentSplitsMenu();
394 					setDefaultWidthColumnsToFit();
395 					getTableHeader().repaint();
396 					clearSelection();
397 				}
398 			}
399 		} else if (action.equals(join)) {
400 			if (popupModelColumn >= 0) {
401 				sprModel.splitColumn(null, popupModelColumn);
402 				refreshRecentSplitsMenu();
403 				setDefaultWidthColumnsToFit();
404 				getTableHeader().repaint();
405 				clearSelection();
406 			}
407 		} else if (action.equals(sortAsc) || action.equals(sortDes)) {
408 			if (popupModelColumn >= 0) {
409 				sprModel.sortRowsByColumn(popupModelColumn);
410 				setDefaultWidthColumnsToFit();
411 				clearSelection();
412 			}
413 		} else if (action.equals(defaultColumnVisibilityString)) {
414 			sprModel.setDefaultColumnVisibility();
415 			refreshPropertiesColumnMenuItemsState();
416 			setDefaultWidthColumnsToFit();
417 			clearSelection();
418 		} else if (action.equals(hideAll) || action.equals(showAll)) {
419 			boolean visible = action.equals(showAll);
420 
421 			int hideableColumnCount = 0;
422 			for (int i = 0; i < sprModel.getPropertiesColumnCount(); i++) {
423 				if (!sprModel.isSolidBaseColumn(i)) {
424 					propertiesColumnMenuItem[i].setSelected(visible);
425 					hideableColumnCount++;
426 				}
427 			}
428 			int[] columns = new int[hideableColumnCount];
429 			int h = 0;
430 			for (int i = 0; i < sprModel.getPropertiesColumnCount(); i++) {
431 				if (!sprModel.isSolidBaseColumn(i)) {
432 					columns[h] = i;
433 					h++;
434 				}
435 			}
436 			sprModel.setPropertyColumnsVisibility(columns, visible);
437 			setDefaultWidthColumnsToFit();
438 			clearSelection();
439 
440 		} else if (action.equals(showAllRowsString)) {
441 			boolean state = ((JCheckBoxMenuItem)source).isSelected();
442 			sprModel.setShowAllRows(state);
443 			refreshShowAllRowsItem();
444 			setDefaultWidthColumnsToFit();
445 			getTableHeader().repaint();
446 			clearSelection();
447 		} else if (action.equals(groupColumnsByGuiGroupString)) {
448 			boolean state = ((JCheckBoxMenuItem)source).isSelected();
449 			sprModel.setGroupColumnsByGuiGroup(state);
450 			refreshShowAllRowsItem();
451 			setDefaultWidthColumnsToFit();
452 			getTableHeader().repaint();
453 			clearSelection();
454 		} else if (action.equals(backgroundColorString)) {
455 			final JColorChooser chooser = ColorChooser.getInstance();
456 
457 			chooser.setColor(getBackground());
458 			ActionListener okListener = new ActionListener() {
459 				public void actionPerformed(ActionEvent event) {
460 					sprModel.setBackground(chooser.getColor());
461 					refreshBackgroundColor();
462 				}
463 			};
464 			JColorChooser.createDialog(inspector, colorChooserTitle, true, chooser, okListener, null)
465 			.setVisible(true); 
466 		} else if (action.equals(defaultColorString)) {
467 			sprModel.setBackground(sprModel.getDefaultBackground());
468 			refreshBackgroundColor();
469 		} else if (action.equals(extendCounters)) {
470 
471 			int[] selViewColumns = getSelectedColumns();
472 			int[] selColumns = new int[selViewColumns.length]; 
473 			for (int c = 0; c < selViewColumns.length; c++) {
474 				selColumns[c] = convertColumnIndexToModel(selViewColumns[c]);
475 			}
476 			
477 			int[] selRows = getSelectedRows();
478 			if (!getRowSelectionAllowed()) {
479 				selRows = new int[sprModel.getRowCount()];
480 				for (int r = 0; r < selRows.length; r++) {
481 					selRows[r] = r;
482 				}
483 			}
484 			sprModel.extendCounters(selRows, selColumns);
485 
486 		} else if (action.equals(hideRow) || action.equals(hideRows)
487 				|| action.equals(showRow) || action.equals(showRows)) {
488 
489 			boolean visible = action.equals(showRow) || action.equals(showRows);
490 
491 			int[] selRows = getSelectedRows();
492 			if (selRows.length == 0) {
493 				selRows = new int[] {popupModelRow};
494 			}
495 			sprModel.setRowsVisibility(selRows, visible);
496 			setDefaultWidthColumnsToFit();
497 			clearSelection();
498 
499 		} else if (action.equals(deleteRow) || action.equals(deleteRows)) {
500 
501 			int[] selRows = getSelectedRows();
502 			if (selRows.length == 0) {
503 				selRows = new int[] {popupModelRow};
504 			}
505 			sprModel.deleteRows(selRows);
506 			clearSelection();
507 
508 		} else if (source instanceof JCheckBoxMenuItem) {
509 			int columnIndex = sprModel.getPropertiesColumnIndex(action);
510 			sprModel.setPropertyColumnsVisibility(new int[] {columnIndex}, ((JCheckBoxMenuItem)source).isSelected());
511 			setDefaultWidthColumnsToFit();
512 			clearSelection();
513 		} else {
514 			if (popupModelColumn >= 0) {
515 				int recentIndex = getRecentSplitDataIndex(action);
516 				if (recentIndex >= 0) {
517 					sprModel.splitColumnByRecentList(recentIndex, popupModelColumn);
518 					refreshRecentSplitsMenu();
519 					setDefaultWidthColumnsToFit();
520 					getTableHeader().repaint();
521 					clearSelection();
522 					return;
523 				}
524 			}
525 
526 			sprModel.setColumnOrder(action);
527 			refreshAll();
528 		}
529 	}

531 	private void addListeners() {
532 
533 		sprModel.setDefaultBackground(getBackground());
534 		sprModel.setBackground(getBackground());
535 		refreshBackgroundColor(); 
536 
537 		getTableHeader().addMouseListener(new MouseAdapter() {
538 
539 			private int draggedColumnModelIndex = -1;
540 			private int draggedColumnViewIndex = -1;
541 			private boolean columnResized = false;
542 
543 			public void mouseClicked(MouseEvent event) {
544 
545 				int columnIndex = columnAtPoint(event.getPoint());
546 				int columnModelIndex = convertColumnIndexToModel(columnIndex);
547 				int posX = event.getX();
548 				int posY = event.getY();
549 				if (event.getButton() == MouseEvent.BUTTON1) {
550 
551 					/* If this is click on the area between columns, set the default column size.
552 					 */ 
553 					Point pointToLeft = new Point(posX - columnMoveMargin, posY);
554 					Point pointToRight = new Point(posX + columnMoveMargin, posY);
555 					int nearbyIndex = columnAtPoint(pointToLeft);
556 					if (nearbyIndex == columnIndex || nearbyIndex == -1) {
557 						nearbyIndex = columnAtPoint(pointToRight);
558 					}
559 					if (nearbyIndex != columnIndex && nearbyIndex >= 0) {
560 						columnIndex = Math.min(columnIndex, nearbyIndex);
561 						setColumnSizeToFit(columnIndex);
562 						sprModel.onColumnWidthChange();
563 					} else {
564 						/* If this is a click on the first column, switch showAllRows mode, otherwise
565 						 * change column sorting if double click.
566 						 */ 
567 						if (columnModelIndex == 0) {
568 							boolean newState = !sprModel.isShowAllRows(); 
569 							sprModel.setShowAllRows(newState);
570 							refreshShowAllRowsItem();
571 							setDefaultWidthColumnsToFit();
572 							getTableHeader().repaint();
573 							clearSelection();
574 							getThis().setColumnSelectionInterval(0, 0);
575 						} else {
576 							if (event.getClickCount() >= 2 && event.getClickCount() % 2 == 0) {
577 								if (columnModelIndex > 0) {
578 									sprModel.sortRowsByColumn(columnModelIndex);
579 									setDefaultWidthColumnsToFit();
580 									clearSelection();
581 									getThis().setColumnSelectionInterval(columnModelIndex, columnModelIndex);
582 								}							
583 							}
584 						}
585 					}
586 				} else if (event.getButton() == MouseEvent.BUTTON3) {
587 					displayPopupMenu(true, getThis(), -1, columnModelIndex, posX, posY);
588 				}
589 			}
590 
591 			public void mousePressed(MouseEvent event) {
592 
593 				JTableHeader header = getTableHeader();
594 
595 				int column = header.columnAtPoint(event.getPoint());
596 				if (column >= 0) {
597                     /* Enable column selections when clicking on the header.
598                      */
599 					setColumnSelectionAllowed(true);
600 					setRowSelectionAllowed(false);
601 					requestFocus();
602 					
603 					// A workaround for a viewport reset during column drag bug with the Java6_03.
604 					setColumnSelectionInterval(column, column);
605 					final Action focusAction = getActionMap().get("focusHeader");
606 					// Older versions have no such actions.
607 					if (focusAction != null) {
608 						focusAction.actionPerformed(new ActionEvent(getThis(), 0, "focusHeader"));
609 					}
610 				}
611 
612 				// Validate dragged column or forbid the drag.
613 				TableColumn draggedColumn = header.getDraggedColumn(); 
614 				if (draggedColumn != null) {
615 					int colIndex = sprModel.validateDraggedColumnIndex(draggedColumn.getModelIndex());
616 					if (colIndex >= 0) {
617 						// Change the dragged column. 
618 						int viewIndex = convertColumnIndexToView(colIndex);
619 						header.setDraggedColumn(header.getColumnModel().getColumn(viewIndex));
620 
621 						draggedColumnModelIndex = colIndex;
622 						draggedColumnViewIndex = viewIndex;
623 					} else {
624 						header.setDraggedColumn(null);
625 					}
626 				}
627 
628 				// Set the column state to non default width if the width is changed by the user.
629 				SpreadsheetColumn resizedColumn = (SpreadsheetColumn)getTableHeader().getResizingColumn();
630 				if (resizedColumn != null) {
631 					resizedColumn.setDefaultWidth(false);
632 					columnResized = true;
633 				}
634 			}
635 
636 			public void mouseReleased(MouseEvent event) {
637 				if (event.getButton() == MouseEvent.BUTTON1) {
638 					// Put the empty column back if it was shifted away. 
639 					int viewIndex = convertColumnIndexToView(0);
640 					if (viewIndex > 0) {
641 						moveColumn(viewIndex, 0);
642 					}
643 
644 					// Validate the positions of columns if there was a drag.
645 					if (draggedColumnModelIndex >= 0) {
646 						int newViewIndex = convertColumnIndexToView(draggedColumnModelIndex);
647 						if (newViewIndex != draggedColumnViewIndex) {
648 							sprModel.repositionColumn(draggedColumnViewIndex, newViewIndex);
649 							setDefaultWidthColumnsToFit();
650 							clearSelection();
651 						}
652 						draggedColumnModelIndex = -1;
653 					}
654 
655 					if (columnResized) {
656 						sprModel.onColumnWidthChange();
657 						columnResized = false;
658 					}
659 				}
660 			}
661 		});
662 
663 		addMouseListener(new MouseAdapter() {
664 			public void mouseClicked(MouseEvent event) {
665 				if (event.getButton() == MouseEvent.BUTTON1 && event.getClickCount() >= 2) {
666 
667 					Point point = event.getPoint();
668 					int row = rowAtPoint(point);
669 					int column = convertColumnIndexToModel(columnAtPoint(point));
670 					String comment = sprModel.getMultilineString(row, column);
671 
672 					if (comment!= null) {
673 						CommentDialog dialog = inspector.getCommentDialog();
674 						dialog.setComment(comment);
675 						dialog.setLocationRelativeTo(inspector);
676 						dialog.setVisible(true);
677 						if (dialog.isConfirmed()) {
678 							setValueAt(dialog.getComment(), row, column);
679 							sprModel.fireTableCellUpdated(row, column);
680 						}
681 					}
682 				}
683 
684 				if (event.getButton() == MouseEvent.BUTTON3) {
685 					Point point = event.getPoint();
686 					int row = rowAtPoint(point);
687 					int column = convertColumnIndexToModel(columnAtPoint(point));
688 					displayPopupMenu(false, getThis(), row, column, event.getX(), event.getY());
689 				}
690 			}
691 		}); 
692 
693 		pane.addMouseListener(new MouseAdapter() {
694 			public void mouseClicked(MouseEvent event) {
695 				if (event.getButton() == MouseEvent.BUTTON3) {
696 					displayPopupMenu(false, getTableScrollPane(), -1, -1, event.getX(), event.getY());
697 				}
698 			}
699 		});
700 	}

702 	private void displayPopupMenu(boolean forHeader, JComponent component, int row, int column, int posX, int posY) {
703 
704 		popupModelRow = row;
705 		popupModelColumn = column;
706 
707 		boolean first = column == 0;
708 		boolean onRow = row >= 0;
709 		boolean onColumn = column >= 0;
710 		boolean mouseOverSelectedRow = false;
711 		if (getSelectedRow() >= 0 && getSelectedColumn() == 0) {
712 			int[] selRows = getSelectedRows();
713 			int r = 0;
714 			for (r = 0; r < selRows.length; r++) {
715 				if (row == selRows[r]) {
716 					break;
717 				}
718 			}
719 			mouseOverSelectedRow = r < selRows.length;
720 		}
721 		/* If there is a row selected or the click is on the first column, bring up a row popup menu, otherwise bring
722 		 * up a column popup menu.
723 		 */
724 		if (mouseOverSelectedRow || (first && onRow)) {
725 			/* Enable quick hidden row switching when all rows are displayed and none selected.
726 			 */
727 			if (mouseOverSelectedRow || !sprModel.isShowAllRows()) {
728 
729 				boolean multipleSelected = getSelectedRowCount() > 1;
730 				boolean visible = false; 
731 				if (!multipleSelected) {
732 					int modelIndex = (getSelectedRowCount() > 0) ? getSelectedRow() : row;
733 					visible = sprModel.isRowVisible(modelIndex);
734 				}
735 
736 				setJComponentVisible(showRowsItem, multipleSelected || !visible);
737 				setJComponentVisible(hideRowsItem, multipleSelected || visible);
738 
739 				showRowsItem.setText(multipleSelected ? showRows : showRow);
740 				hideRowsItem.setText(multipleSelected ? hideRows : hideRow);
741 				deleteRowsItem.setText(multipleSelected ? deleteRows : deleteRow);
742 
743 				rowPopupMenu.show(component, posX, posY);
744 			} else {
745 				boolean visible = sprModel.isRowVisible(row);
746 				sprModel.setRowsVisibility(new int[] {row}, !visible);
747 				setDefaultWidthColumnsToFit();
748 				clearSelection();
749 			}
750 
751 		} else {
752 			boolean onTableOrHeader = (forHeader || onRow) && onColumn;
753 			boolean persistant = onColumn && sprModel.isSolidColumn(column);
754 			boolean splitted = onTableOrHeader && sprModel.isSplit(column);
755 			boolean manipulatable = onTableOrHeader && !first && !persistant; 
756 			boolean sortable = onTableOrHeader && !first;
757 
758 			setJComponentVisible(hideItem, manipulatable);
759 			setJComponentVisible(recentSplitsMenu, manipulatable && !splitted);
760 			setJComponentVisible(joinItem, manipulatable && splitted);
761 			setJComponentVisible(sortAscItem, sortable);
762 			setJComponentVisible(sortDesItem, sortable);
763 			setJComponentVisible(sortSeparator, sortable);
764 
765 			boolean extendCountersEnabled = false;
766 			if (getRowSelectionAllowed()) {
767 				extendCountersEnabled = getSelectedColumnCount() >= 1 && getSelectedRowCount() >= 2;
768 			} else {
769 				extendCountersEnabled = getSelectedColumnCount() >= 1 && sprModel.getRowCount() >= 2;
770 			}
771 			extendCountersItem.setEnabled(extendCountersEnabled);
772 			
773 			columnPopupMenu.show(component, posX, posY);
774 		}
775 	}
