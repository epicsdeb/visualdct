29  package com.cosylab.vdct.inspector.sheet;
30  
31  import java.awt.Dialog;
32  import java.awt.Dimension;
33  import java.awt.GridBagConstraints;
34  import java.awt.GridBagLayout;
35  import java.awt.Insets;
36  import java.awt.event.ActionEvent;
37  import java.awt.event.ActionListener;
38  
39  import javax.swing.JButton;
40  import javax.swing.JDialog;
41  import javax.swing.JLabel;
42  import javax.swing.JPanel;
43  import javax.swing.JScrollPane;
44  import javax.swing.JTextArea;
45  import javax.swing.WindowConstants;
46  
47  /**
48   * @author ssah
49   */
50  public class CommentDialog extends JDialog implements ActionListener {
51  
52  	private JTextArea textArea = null;
53  	private boolean confirmed = true;
54  
55  	private static final String okString = "OK";
56  	private static final String cancelString = "Cancel";
57  
58  	public CommentDialog(Dialog dialog) {
59          super(dialog, true);
60  		setTitle("Comment");
61          setDefaultCloseOperation(WindowConstants.HIDE_ON_CLOSE);
62          createGUI();
63      }
64  	
65  	/** This method is called from within the constructor to initialize the form.
66       */
67      private void createGUI() {
68  
69  		JPanel contentPanel = new JPanel(new GridBagLayout());
70  		contentPanel.setPreferredSize(new Dimension(320, 240));
71  		
72  		JLabel dummyLabel = new JLabel();
73  		textArea = new JTextArea();
74  		textArea.setFont(dummyLabel.getFont());
75  		
76  		JScrollPane scrollPane = new JScrollPane();
77  		scrollPane.setViewportView(textArea);
78  		GridBagConstraints constraints = new GridBagConstraints();
79  		constraints.weightx = 1.0;
80  		constraints.weighty = 1.0;
81  		constraints.fill = GridBagConstraints.BOTH;
82  		constraints.insets = new Insets(4, 4, 4, 4);
83  		contentPanel.add(scrollPane, constraints);
84  
85  	    JPanel buttonPanel = createButtonPanel();
86  		constraints = new GridBagConstraints();
87  		constraints.gridy = 1;
88  		constraints.weightx = 1.0;
89  		constraints.fill = GridBagConstraints.HORIZONTAL;
90          constraints.insets = new Insets(4, 4, 4, 4);
91          contentPanel.add(buttonPanel, constraints);
92          
93  		getContentPane().add(contentPanel);
94  		pack();
95      }
96  
97      private JPanel createButtonPanel() {
98      	JPanel buttonPanel = new JPanel(new GridBagLayout());
99  		
100     	JButton okButton = new JButton(okString); 
101     	okButton.setMnemonic('O');
102     	okButton.addActionListener(this);
103 		GridBagConstraints constraints = new GridBagConstraints();
104 		constraints.weightx = .5;
105 		constraints.anchor = GridBagConstraints.EAST;
106         constraints.insets = new Insets(4, 4, 4, 4);
107         buttonPanel.add(okButton, constraints);
108 
109     	JButton cancelButton = new JButton(cancelString); 
110     	cancelButton.setMnemonic('C');
111     	cancelButton.addActionListener(this);
112     	
113 		constraints = new GridBagConstraints();
114 		constraints.gridx = 1;
115 		constraints.weightx = .5;
116 		constraints.anchor = GridBagConstraints.WEST;
117         constraints.insets = new Insets(4, 4, 4, 4);
118         buttonPanel.add(cancelButton, constraints);
119         
120         return buttonPanel;
121     }
122     
123     public void setComment(String comment) {
124     	textArea.setText(comment);
125     }
126 
127     public String getComment() {
128     	return textArea.getText();
129     }
130 
131 	public void actionPerformed(ActionEvent event) {
132 		confirmed = event.getActionCommand().equals(okString);
133        	setVisible(false);
134 	}
135 
136 	/**
137 	 * @return the confirmed
138 	 */
139 	public boolean isConfirmed() {
140 		return confirmed;
141 	}
142 }
