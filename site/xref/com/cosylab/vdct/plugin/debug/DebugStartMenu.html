84  public void actionPerformed(ActionEvent event)
85  {
86  	
87  	rootGroups = Group.getAllRoots();
88  	if (plugin != null)
89  	{
90  
91  		if (PluginDebugManager.getDebugPlugin()!=null)
92  		{
93  			Console.getInstance().println("Debug pluging '" + PluginDebugManager.getDebugPlugin().getName()+ "' is already running, stop it first...");
94  			Toolkit.getDefaultToolkit().beep();
95  			return;
96  		}
97  		
98  		DebugPlugin debugPlugin = (DebugPlugin)plugin.getPlugin();
99  
100 		Console.getInstance().println("Starting debugging with '" + debugPlugin.getName()+ "'...");
101 
102 		PluginDebugManager.setDebugPlugin(debugPlugin);
103 		PluginDebugManager.setDebugState(true);
104 		debugPlugin.startDebugging();
105 
106 	
107 		new Thread(this).start();
108 	}
109 }

111 /**
112  * Thread worker connecting to the PVs.
113  * @see java.lang.Runnable#run()
114  */
115 public void run()
116 {
117 	DebugPlugin debugPlugin = PluginDebugManager.getDebugPlugin();
118 	
119 
120 	// all new (or deleted) filed are not updated
121 	// if field has InspectableProperty.ALWAYS_VISIBLE visibility then is is monitored
122 	// VAL fields is always registered
123 	final String valFieldName = "VAL";
124 	
125 	// count loop
126 	int recordCount = 0;
127 	Stack groupStack = new Stack();
128 	
129 	Iterator iterator = rootGroups.iterator();
130 	while (iterator.hasNext()) {
131 	    groupStack.push(iterator.next());
132 	}
133 
134 	while (!groupStack.isEmpty())
135 	{
136 		Group group = (Group)groupStack.pop();
137 		Enumeration e = group.getSubObjectsV().elements();
138 		while (e.hasMoreElements())
139 		{
140 			Object obj = e.nextElement();
141 			if (obj instanceof Record)
142 				recordCount++;
143 			else if (obj instanceof Group)
144 				groupStack.push(obj);
145 		}
146 	}
147 
148 
149 	// connect loop
150 	iterator = rootGroups.iterator();
151 	while (iterator.hasNext()) {
152 	    groupStack.push(iterator.next());
153 	}
154 
155 	int progress = 0;
156 	ProgressMonitor progressMonitor = new ProgressMonitor(VisualDCT.getInstance(),
157 								"Connecting to PVs",
158 								"Initializing...", 0, recordCount);
159 	progressMonitor.setProgress(0);
160 
161 	while (!groupStack.isEmpty())
162 	{
163 
164 		Group group = (Group)groupStack.pop();
165 		Enumeration e = group.getSubObjectsV().elements();
166 		while (e.hasMoreElements() && !progressMonitor.isCanceled())
167 		{
168 			Object obj = e.nextElement();
169 			if (obj instanceof Record)
170 			{
171 				VDBRecordData rec = ((Record)obj).getRecordData();
172 				Enumeration e2 = rec.getFieldsV().elements();
173 				while (e2.hasMoreElements() && !progressMonitor.isCanceled() && PluginDebugManager.isDebugState())
174 				{
175 					VDBFieldData field = (VDBFieldData)e2.nextElement();
176 						
177 					if (field.getVisibility() == InspectableProperty.ALWAYS_VISIBLE && !field.equals(valFieldName))
178 					{
179 						progressMonitor.setNote("Connecting to "+field.getFullName()+"...");
180 						debugPlugin.registerMonitor(field);
181 					}
182 				}
183 	
184 				if (progressMonitor.isCanceled() || !PluginDebugManager.isDebugState())
185 					break;
186 	
187 				// always register VAL field
188 				Debuggable valField = (Debuggable)rec.getField(valFieldName);
189 				if (valField != null)
190 				{
191 					progressMonitor.setNote("Connecting to "+valField.getFullName()+"...");
192 					debugPlugin.registerMonitor(valField);
193 				}
194 						
195 			}
196 			else if (obj instanceof Group)
197 				groupStack.push(obj);
198 				
199 			progress++;
200 			progressMonitor.setProgress(progress);
201 				
202 		}
203 	}
204 	
205 			
206 	if (progressMonitor.isCanceled() /*|| !PluginDebugManager.isDebugState()*/)
207 	{
208 		Console.getInstance().println("Debugging canceled.");
209 		debugPlugin.deregisterAll();
210 		debugPlugin.stopDebugging();
211 		PluginDebugManager.setDebugState(false);
212 		PluginDebugManager.setDebugPlugin(null);
213 	}
214 
215 	progressMonitor.close();
216 
217 	CommandManager.getInstance().execute("RepaintAllFrames");
218 }

318 /**
319  * Insert the method's description here.
320  * Creation date: (7.12.2001 17:09:23)
321  * @param plugin */
322 public void pluginAdded(PluginObject plugin)
323 {
324 	if (plugin.getPlugin() instanceof DebugPlugin)
325 	{
326 		if( plugin.getStatus()==PluginObject.PLUGIN_NOT_LOADED ||
327 		    plugin.getStatus()==PluginObject.PLUGIN_INVALID )
328 			    return;
329 
330 		DebugPluginMenuItem menu = new DebugPluginMenuItem(plugin);
331 
332 		add(menu);
333 		exportMenuItems.put(plugin, menu);
334 
335 		if (getItemCount()>0)
336 			setEnabled(true);
337 	}
338 
339 }
340 /**
341  * Insert the method's description here.
342  * Creation date: (7.12.2001 17:10:37)
343  * @param plugin */
344 public void pluginRemoved(PluginObject plugin)
345 {
346 	if (plugin.getPlugin() instanceof DebugPlugin)
347 	{
348 		DebugPluginMenuItem menuItem = (DebugPluginMenuItem)exportMenuItems.remove(plugin);
349 
350 		if (menuItem!=null)
351 		{
352 			remove(menuItem);
353 			menuItem.setPlugin(null);
354 
355 			if (getItemCount()==0)
356 				setEnabled(false);
357 		}
358 	}
359 }
