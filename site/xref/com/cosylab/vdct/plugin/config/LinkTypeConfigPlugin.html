40  public interface LinkTypeConfigPlugin extends Plugin {
41  
42  /**
43   * This method should return a table of different link type configurations.
44   * Each element of the list should have a key identifiying the link type (e.g. VME_IO) and
45   * have the value of type Object[3], where array consists of
46   * { java.util.regex.Pattern, java.lang.String, java.lang.String }.
47   * The array represents { link validation schema, default schema, description }.
48   * An example of default implementation:
49   * <pre>
50   * 
51   * 	Hashtable linkTypeConfigTable = new Hashtable();
52   * 
53   *	linkTypeConfigTable.put("CONSTANT", new Object[] { Pattern.compile(".*"), "", "CONSTANT" });
54   *	linkTypeConfigTable.put("PV_LINK", new Object[] { Pattern.compile(".*"), "", "PV_LINK" } );
55   *	linkTypeConfigTable.put("VME_IO", new Object[] { Pattern.compile("#C\\d+ S\\d+ @.*"), "#C0 S0 @", "VME_IO - #Ccard Ssignal @parm" });
56   *	linkTypeConfigTable.put("CAMAC_IO", new Object[] { Pattern.compile("#B\\d+ C\\d+ N\\d+ A\\d+ F\\d+ @.*"), "CAMAC_IO - #B0 C0 N0 A0 F0 @", "#Bbranch Ccrate Nstation Asubaddress Ffunction @parm" });
57   *	linkTypeConfigTable.put("AB_IO", new Object[] { Pattern.compile("#L\\d+ A\\d+ C\\d+ S\\d+ @.*"), "#L0 A0 C0 S0 @", "AB_IO - #Llink Aadapter Ccard Ssignal @parm" });
58   *	linkTypeConfigTable.put("GPIB_IO", new Object[] { Pattern.compile("#L\\d+ A\\d+ @.*"), "#L0 A0 @", "GPIB_IO - #Llink Aaddr @parm" });
59   *	linkTypeConfigTable.put("BITBUS_IO", new Object[] { Pattern.compile("#L\\d+ N\\d+ P\\d+ S\\d+ @.*"), "BITBUS_IO - @L0 N0 P0 S0 @", "#Llink Nnode Pport Ssignal @parm" });
60   *	linkTypeConfigTable.put("INST_IO", new Object[] { Pattern.compile("@.*"), "@", "INST_IO - @" });
61   *	linkTypeConfigTable.put("RF_IO", new Object[] { Pattern.compile("#R\\d+ M\\d+ D\\d+ E\\d+ @.*"), "#R0 M0 D0 E0 @", "RF_IO - #Rcryo Mmicro Ddataset Eelement" });
62   *	linkTypeConfigTable.put("VXI_IO", new Object[] { Pattern.compile("#V\\d+ (C\\d+)?+  S\\d+ @.*"), "#V0 C0 S0 @", "VXI_IO - #Vframe Cslot Ssignal @parm" });
63   * </pre> 
64   * If <code>null</code> is returned, VisualDCT interprets this as 'failed to load' / 'no configuration found'. 
65   * If all <code>LinkTypeConfigPlugin</code> plugins return <code>null</code>, VisualDCT loads default configuration.
66   * Plugins are loaded as defined in VisualDCT plugin configuration. Configurations are additive, i.e. if there are two configurations for the same link type, the last overrides the first.   
67   * Creation date: (8.12.2001 12:45:31)
68   * @return java.util.ArrayList
69   */
70  public Hashtable getLinkTypeConfig();
71  }
