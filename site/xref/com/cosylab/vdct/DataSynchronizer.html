48  /**
49   * <code>DataSynchronizer</code> handles synchronization between loaded files and files on the file
50   * system. It checks for unsaved changes or file system changes and brings up the appropriate dialogs.    
51   * 
52   * @author ssah
53   */
54  public class DataSynchronizer {
55  
56  	private static DataSynchronizer instance = null;
57  	private Component dialogParent = null;
58  	private DsManagerInterface dsManager = null;
59  
60      /**
61       *
62       * @return something
63       */
64      public static DataSynchronizer getInstance() {
65  		if (instance == null) {
66  			instance = new DataSynchronizer();
67  		}
68  		return instance;
69  	}
70  
71      /**
72       *
73       */
74      public DataSynchronizer() {
75  		super();
76  		GetMainComponent command = (GetMainComponent)CommandManager.getInstance().getCommand("GetMainComponent");
77  		if (command != null) {
78  		    dialogParent = command.getComponent();
79  		}
80  		dsManager = ((GetDsManager)CommandManager.getInstance().getCommand("GetDsManager")).getManager();
81  	}

83      /**
84       *
85       * @param dsId dsId
86       * @param templateId templateId
87       * @return something
88       */
89      public VDBTemplate getTemplate(Object dsId, String templateId) {
90  
91  		VDBTemplate template = (VDBTemplate)VDBData.getInstance(dsId).getTemplates().get(templateId);
92  		if (template == null) {
93  			// Try to find template in another drawing surface.
94  			DrawingSurfaceInterface[] surfaces = dsManager.getDrawingSurfaces();
95  			for (int i = 0; i < surfaces.length && template == null; i++) {
96  				template = (VDBTemplate)VDBData.getInstance(surfaces[i].getDsId()).getTemplates().get(templateId);
97  			}
98  			if (template != null) {
99  				File file = new File(template.getFileName());
100 				boolean loadSuccessful = false;
101 				try {
102 					loadSuccessful = dsManager.getDrawingSurfaceById(dsId).open(file, true);
103 				} catch (Exception exception) {
104 					Console.getInstance().println("Failed to load template file '" + template.getFileName() + "'.");	
105 				}
106 				if (loadSuccessful) {
107 					template = (VDBTemplate)VDBData.getInstance(dsId).getTemplates().get(templateId);
108 				}
109 			}
110 		}
111 		if (template == null) {
112 			Console.getInstance().println("Could not load template '" + templateId + "'.");
113 		}
114 		return template;
115 	}

117     /**
118      *
119      * @param dsId dsId
120      * @param exit exit
121      * @return something
122      */
123     public boolean confirmFileClose(Object dsId, boolean exit) {
124 		boolean confirmed = true;
125 		if (dsId != null) {
126 			if (confirmUnsavedChangesDialog(dsManager.getDrawingSurfaceById(dsId))) {
127 				showChangedMacrosPortsDialog(dsId);
128 			} else {
129 				confirmed = false;
130 			}
131 		} else {
132 			DrawingSurfaceInterface[] surfaces = dsManager.getDrawingSurfaces();
133 			for (int i = 0; i < surfaces.length && confirmed; i++) {
134 				confirmed = confirmUnsavedChangesDialog(surfaces[i]);
135 			}
136 
137 			if (confirmed) {
138 				surfaces = dsManager.getDrawingSurfaces();
139 				String changedTemplateTitles = new String();
140 				for (int i = 0; i < surfaces.length && confirmed; i++) {
141 					if (surfaces[i].isTemplateChanged()) {
142 						changedTemplateTitles = changedTemplateTitles + surfaces[i].getTitle() + "\n";
143 					}
144 				}
145 
146 				if (changedTemplateTitles.length() > 0) {
147 					String description = "Macros and/or ports in the following templates have changed.\n" + changedTemplateTitles + "Reload and save files that include these templates to apply changes.";
148 					String title = "Templates changed!";
149 
150 					if (exit) {
151 						int selection = JOptionPane.showConfirmDialog(
152 								dialogParent,
153 								description + "\n\nAre you sure you want to exit VisualDCT?\n",
154 								title, JOptionPane.YES_NO_OPTION,
155 								JOptionPane.WARNING_MESSAGE);
156 						confirmed = selection == JOptionPane.YES_OPTION;
157 					} else {
158 						JOptionPane.showMessageDialog(dialogParent,
159 								description, title, JOptionPane.WARNING_MESSAGE);
160 					} 
161 				}
162 			}
163 		}
164 		return confirmed;
165 	}

167     /**
168      *
169      * @param dsId dsId
170      */
171     public void checkFilesystemChanges(Object dsId) {
172 
173 		VDBTemplate template = null; 
174 
175 		if (dsId != null) {
176 			template = dsManager.getDrawingSurfaceById(dsId).getTemplate();
177 		}
178 
179 		DrawingSurfaceInterface[] surfaces = dsManager.getDrawingSurfaces();
180 		for (int i = 0; i < surfaces.length; i++) {
181 			Object id = surfaces[i].getDsId();
182 			template = dsManager.getDrawingSurfaceById(id).getTemplate();
183 
184 			if (template != null && newerFileExists(template)) {
185 				dsManager.setFocusedDrawingSurface(id);
186 				int selection = JOptionPane.showConfirmDialog(dialogParent,
187 						"There is a newer file on the file system. Reload?",
188 						surfaces[i].getTitle(), JOptionPane.YES_NO_OPTION, JOptionPane.WARNING_MESSAGE);
189 				if (selection == JOptionPane.YES_OPTION) {
190 					String fileName = template.getFileName();
191 					boolean success = false;
192 					try {
193 						success = surfaces[i].open(new File(fileName), false);
194 					} catch (IOException exception) {
195 						Console.getInstance().println(exception);
196 					}
197 					if (!success) {
198 						Console.getInstance().println("Failed to reload '" + fileName + "'.");
199 					}
200 				}
201 			}
202 
203 			Vector templatesVector = new Vector(VDBData.getInstance(id).getTemplates().values());
204 			Iterator templatesIterator = templatesVector.iterator();
205 			while (templatesIterator.hasNext()) {
206 				template = (VDBTemplate)templatesIterator.next();
207 				if (newerFileExists(template)) {
208 					surfaces[i].reloadTemplate(template);
209 				}
210 			}
211 		}
212 	}

214 	private boolean confirmUnsavedChangesDialog(DrawingSurfaceInterface drawingSurface) {
215 
216 		boolean confirmed = true;
217 
218 		if (drawingSurface != null && drawingSurface.isModified()) {
219 			dsManager.setFocusedDrawingSurface(drawingSurface.getDsId());
220 			int selection = JOptionPane.showConfirmDialog(dialogParent, "The file has been modified. Save changes?",
221 					drawingSurface.getTitle(), JOptionPane.YES_NO_CANCEL_OPTION, JOptionPane.WARNING_MESSAGE);
222 			confirmed = selection != JOptionPane.YES_OPTION;
223 
224 			if (selection == JOptionPane.YES_OPTION) {
225 
226 				dialogParent = ((GetMainComponent)CommandManager.getInstance().getCommand("GetMainComponent")).getComponent();
227 
228 				SaveCommand saveCommand = (SaveCommand)CommandManager.getInstance().getCommand("SaveCommand");
229 				saveCommand.execute();
230 				confirmed = saveCommand.isSuccess();
231 			} else if (selection == JOptionPane.NO_OPTION) {
232 			} else {
233 				confirmed = false;
234 			}
235 		}
236 		return confirmed;
237 	}
238 
239 	private void showChangedMacrosPortsDialog(Object dsId) {
240 		if (dsManager.getDrawingSurfaceById(dsId).isTemplateChanged()) {
241 			dsManager.setFocusedDrawingSurface(dsId);
242 			JOptionPane.showMessageDialog(dialogParent,
243 					"Macros and/or ports in this template have changed. \nReload and save files that include this template to apply changes.",
244 					"Template '" + dsManager.getDrawingSurfaceById(dsId).getTitle() + "' changed!",
245 					JOptionPane.WARNING_MESSAGE);
246 		}
247 	}
248 
249 	private boolean newerFileExists(VDBTemplate template) {
250 		long storedModificationTime = template.getModificationTime();
251 		long currentModificationTime = new File(template.getFileName()).lastModified();
252 
253 		// set the modified date so that the dialog is brought up at most once per file system change
254 		template.setModificationTime(currentModificationTime);
255 		return storedModificationTime != 0 && storedModificationTime < currentModificationTime;
256 	}
