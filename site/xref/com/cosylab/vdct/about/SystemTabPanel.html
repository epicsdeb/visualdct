378 	/**
379 	 * @see com.cosylab.vdct.about.AboutTabPanel#processData()
380 	 */
381 	protected void processData() {
382 		java.util.Properties properties = ((SystemTabModel)model).getSystemProperties();
383 		JLabelSystemJavaProperty.setText(properties.getProperty("java.runtime.name")+" "+properties.getProperty("java.runtime.version"));
384 		JLabelSystemOSProperty.setText(" "+properties.getProperty("os.name")+" "+properties.getProperty("os.version")+
385 									   " on "+properties.getProperty("os.arch"));
386 		
387 		TreeMap tree = new TreeMap(properties);
388 		Iterator i = tree.keySet().iterator();
389 		StringBuffer text = new StringBuffer();
390 		while (i.hasNext())
391 		{
392 			String key = i.next().toString();
393 			String value = properties.get(key).toString();
394 			text.append(key);	
395 			text.append("=");	
396 			text.append(value);	
397 			text.append("\n");	
398 		}
399 		
400 		JTextAreaSystemJavaClasspathProperty.setText(text.toString());
401 		JTextAreaSystemJavaClasspathProperty.setCaretPosition(0);
402 	}
